9impl<const BITS: usize, const LIMBS: usize> Serialize for Signed<BITS, LIMBS> {
10    fn serialize<S: Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
11        serializer.collect_str(self)
12    }

15impl<'de, const BITS: usize, const LIMBS: usize> Deserialize<'de> for Signed<BITS, LIMBS> {
16    fn deserialize<D: Deserializer<'de>>(deserializer: D) -> Result<Self, D::Error> {
17        struct SignedVisitor<const BITS: usize, const LIMBS: usize>;
18
19        impl<const BITS: usize, const LIMBS: usize> Visitor<'_> for SignedVisitor<BITS, LIMBS> {
20            type Value = Signed<BITS, LIMBS>;
21
22            fn expecting(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
23                write!(f, "a {BITS} bit signed integer")
24            }
25
26            fn visit_u64<E: de::Error>(self, v: u64) -> Result<Self::Value, E> {
27                Signed::try_from(v).map_err(de::Error::custom)
28            }
29
30            fn visit_u128<E: de::Error>(self, v: u128) -> Result<Self::Value, E> {
31                Signed::try_from(v).map_err(de::Error::custom)
32            }
33
34            fn visit_i64<E: de::Error>(self, v: i64) -> Result<Self::Value, E> {
35                Signed::try_from(v).map_err(de::Error::custom)
36            }
37
38            fn visit_i128<E: de::Error>(self, v: i128) -> Result<Self::Value, E> {
39                Signed::try_from(v).map_err(de::Error::custom)
40            }
41
42            fn visit_str<E: de::Error>(self, v: &str) -> Result<Self::Value, E> {
43                v.parse().map_err(serde::de::Error::custom)
44            }
45
46            fn visit_string<E: de::Error>(self, v: String) -> Result<Self::Value, E> {
47                self.visit_str(&v)
48            }
49        }
50
51        deserializer.deserialize_any(SignedVisitor)
52    }
53}
