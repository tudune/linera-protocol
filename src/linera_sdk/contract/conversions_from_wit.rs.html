4//! Conversions from types generated by [`wit-bindgen`] to types declared in [`linera-sdk`].
5

15};

22impl From<wit_contract_api::CryptoHash> for CryptoHash {
23    fn from(crypto_hash: wit_contract_api::CryptoHash) -> Self {
24        CryptoHash::from([
25            crypto_hash.part1,
26            crypto_hash.part2,
27            crypto_hash.part3,
28            crypto_hash.part4,
29        ])
30    }
31}
32
33impl From<wit_contract_api::Array20> for [u8; 20] {
34    fn from(ethereum_address: wit_contract_api::Array20) -> Self {
35        let mut bytes = [0u8; 20];
36        bytes[0..8].copy_from_slice(&ethereum_address.part1.to_le_bytes());
37        bytes[8..16].copy_from_slice(&ethereum_address.part2.to_le_bytes());
38        bytes[16..20].copy_from_slice(&ethereum_address.part3.to_le_bytes());
39        bytes
40    }
41}

43impl From<wit_contract_api::AccountOwner> for AccountOwner {
44    fn from(account_owner: wit_contract_api::AccountOwner) -> Self {
45        match account_owner {
46            wit_contract_api::AccountOwner::Reserved(value) => AccountOwner::Reserved(value),
47            wit_contract_api::AccountOwner::Address32(value) => {
48                AccountOwner::Address32(value.into())
49            }
50            wit_contract_api::AccountOwner::Address20(value) => {
51                AccountOwner::Address20(value.into())
52            }
53        }
54    }
55}

57impl From<wit_contract_api::ModuleId> for ModuleId {
58    fn from(module_id: wit_contract_api::ModuleId) -> Self {
59        ModuleId::new(
60            module_id.contract_blob_hash.into(),
61            module_id.service_blob_hash.into(),
62            module_id.vm_runtime.into(),
63        )
64    }
65}
66
67impl From<wit_contract_api::VmRuntime> for VmRuntime {
68    fn from(vm_runtime: wit_contract_api::VmRuntime) -> Self {
69        match vm_runtime {
70            wit_contract_api::VmRuntime::Wasm => VmRuntime::Wasm,
71            wit_contract_api::VmRuntime::Evm => VmRuntime::Evm,
72        }
73    }
74}
75
76impl From<wit_contract_api::MessageId> for MessageId {
77    fn from(message_id: wit_contract_api::MessageId) -> Self {
78        MessageId {
79            chain_id: message_id.chain_id.into(),
80            height: BlockHeight(message_id.height.inner0),
81            index: message_id.index,
82        }
83    }
84}

86impl From<wit_contract_api::ApplicationId> for ApplicationId {
87    fn from(application_id: wit_contract_api::ApplicationId) -> Self {
88        ApplicationId::new(application_id.application_description_hash.into())
89    }

92impl From<wit_contract_api::ChainId> for ChainId {
93    fn from(chain_id: wit_contract_api::ChainId) -> Self {
94        ChainId(chain_id.inner0.into())
95    }
96}

98impl From<wit_contract_api::Amount> for Amount {
99    fn from(balance: wit_contract_api::Amount) -> Self {
100        let (lower_half, upper_half) = balance.inner0;
101        let value = ((upper_half as u128) << 64) | (lower_half as u128);
102        Amount::from_attos(value)
103    }
104}

106impl From<wit_contract_api::CloseChainError> for CloseChainError {
107    fn from(guest: wit_contract_api::CloseChainError) -> Self {
108        match guest {
109            wit_contract_api::CloseChainError::NotPermitted => CloseChainError::NotPermitted,
110        }
111    }
112}

143impl From<wit_entrypoints::GenericApplicationId> for GenericApplicationId {
144    fn from(generic_application_id: wit_entrypoints::GenericApplicationId) -> Self {
145        match generic_application_id {
146            wit_entrypoints::GenericApplicationId::System => GenericApplicationId::System,
147            wit_entrypoints::GenericApplicationId::User(application_id) => {
148                GenericApplicationId::User(application_id.into())
149            }
150        }
151    }
