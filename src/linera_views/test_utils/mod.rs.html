33/// Takes a random number generator, a `key_prefix` and extends it by n random bytes.
34pub fn get_random_byte_vector<R: Rng>(rng: &mut R, key_prefix: &[u8], n: usize) -> Vec<u8> {
35    let mut v = key_prefix.to_vec();
36    for _ in 0..n {
37        let val = rng.gen_range(0..256) as u8;
38        v.push(val);
39    }
40    v
41}

43/// Appends a small value to a key making collisions likely.
44pub fn get_small_key_space<R: Rng>(rng: &mut R, key_prefix: &[u8], n: usize) -> Vec<u8> {
45    let mut key = key_prefix.to_vec();
46    for _ in 0..n {
47        let byte = rng.gen_range(0..4) as u8;
48        key.push(byte);
49    }
50    key
51}

53/// Builds a random k element subset of n
54pub fn get_random_kset<R: Rng>(rng: &mut R, n: usize, k: usize) -> Vec<usize> {
55    let mut values = Vec::new();
56    for u in 0..n {
57        values.push(u);
58    }
59    values.shuffle(rng);
60    values[..k].to_vec()
61}

63/// Takes a random number generator, a `key_prefix` and generates
64/// pairs `(key, value)` with key obtained by appending 8 bytes at random to `key_prefix`
65/// and value obtained by appending 8 bytes to the trivial vector.
66/// We return n such `(key, value)` pairs which are all distinct
67pub fn get_random_key_values_prefix<R: Rng>(
68    rng: &mut R,
69    key_prefix: Vec<u8>,
70    len_key: usize,
71    len_value: usize,
72    num_entries: usize,
73) -> Vec<(Vec<u8>, Vec<u8>)> {
74    loop {
75        let mut v_ret = Vec::new();
76        let mut vector_set = HashSet::new();
77        for _ in 0..num_entries {
78            let v1 = get_random_byte_vector(rng, &key_prefix, len_key);
79            let v2 = get_random_byte_vector(rng, &Vec::new(), len_value);
80            let v12 = (v1.clone(), v2);
81            vector_set.insert(v1);
82            v_ret.push(v12);
83        }
84        if vector_set.len() == num_entries {
85            return v_ret;
86        }
87    }
88}

108/// A random reordering of the puts and deletes.
109/// For something like `MapView` it should get us the same result whatever way we are calling.
110pub fn span_random_reordering_put_delete<R: Rng>(
111    rng: &mut R,
112    info_op: VectorPutDelete,
113) -> Vec<WriteOperation> {
114    let n = info_op.0.len();
115    let k = info_op.1;
116    let mut indices = Vec::new();
117    for i in 0..n {
118        indices.push(i);
119    }
120    indices.shuffle(rng);
121    let mut indices_rev = vec![0; n];
122    for i in 0..n {
123        indices_rev[indices[i]] = i;
124    }
125    let mut pos_remove_vector = vec![Vec::new(); n];
126    for (i, pos) in indices_rev.iter().enumerate().take(k) {
127        let idx = rng.gen_range(*pos..n);
128        pos_remove_vector[idx].push(i);
129    }
130    let mut operations = Vec::new();
131    for i in 0..n {
132        let pos = indices[i];
133        let pair = info_op.0[pos].clone();
134        operations.push(Put {
135            key: pair.0,
136            value: pair.1,
137        });
138        for pos_remove in pos_remove_vector[i].clone() {
139            let key = info_op.0[pos_remove].0.clone();
140            operations.push(Delete { key });
141        }
142    }
143    operations
144}

146/// This test starts with a collection of key/values being inserted into the code
147/// which is then followed by a number of reading tests. The functionalities being
148/// tested are all the reading functionalities:
149/// * `read_value_bytes`
150/// * `read_multi_values_bytes`
151/// * `find_keys_by_prefix` / `find_key_values_by_prefix`
152/// * The ordering of keys returned by `find_keys_by_prefix` and `find_key_values_by_prefix`
153pub async fn run_reads<S: RestrictedKeyValueStore>(store: S, key_values: Vec<(Vec<u8>, Vec<u8>)>) {
154    // We need a nontrivial key_prefix because dynamo requires a non-trivial prefix
155    let mut batch = Batch::new();
156    let mut keys = Vec::new();
157    let mut set_keys = HashSet::new();
158    for (key, value) in &key_values {
159        keys.push(&key[..]);
160        set_keys.insert(&key[..]);
161        batch.put_key_value_bytes(key.clone(), value.clone());
162    }
163    store.write_batch(batch).await.unwrap();
164    for key_prefix in keys
165        .iter()
166        .flat_map(|key| (0..key.len()).map(|u| &key[..=u]))
167    {
168        // Getting the find_keys_by_prefix / find_key_values_by_prefix
169        let len_prefix = key_prefix.len();
170        let keys_by_prefix = store.find_keys_by_prefix(key_prefix).await.unwrap();
171        let keys_request = keys_by_prefix
172            .iterator()
173            .map(Result::unwrap)
174            .collect::<Vec<_>>();
175        let mut set_key_value1 = HashSet::new();
176        let mut keys_request_deriv = Vec::new();
177        let key_values_by_prefix = store.find_key_values_by_prefix(key_prefix).await.unwrap();
178        for (key, value) in key_values_by_prefix.iterator().map(Result::unwrap) {
179            set_key_value1.insert((key, value));
180            keys_request_deriv.push(key);
181        }
182        // Check find_keys / find_key_values
183        assert_eq!(keys_request, keys_request_deriv);
184        // Check key ordering
185        for i in 1..keys_request.len() {
186            assert!(keys_request[i - 1] < keys_request[i]);
187        }
188        // Check the obtained values
189        let mut set_key_value2 = HashSet::new();
190        for (key, value) in &key_values {
191            if key.starts_with(key_prefix) {
192                set_key_value2.insert((&key[len_prefix..], &value[..]));
193            }
194        }
195        assert_eq!(set_key_value1, set_key_value2);
196    }
197    // Now checking the read_multi_values_bytes
198    let mut rng = make_deterministic_rng();
199    for _ in 0..3 {
200        let mut keys = Vec::new();
201        let mut values = Vec::new();
202        for (key, value) in &key_values {
203            if rng.gen() {
204                // Put a key that is already present
205                keys.push(key.clone());
206                values.push(Some(value.clone()));
207            } else {
208                // Put a missing key
209                let len = key.len();
210                let pos = rng.gen_range(0..len);
211                let byte = *key.get(pos).unwrap();
212                let new_byte: u8 = if byte < 255 { byte + 1 } else { byte - 1 };
213                let mut new_key = key.clone();
214                *new_key.get_mut(pos).unwrap() = new_byte;
215                if !set_keys.contains(&*new_key) {
216                    keys.push(new_key);
217                    values.push(None);
218                }
219            }
220        }
221        let mut test_exists = Vec::new();
222        let mut values_single_read = Vec::new();
223        for key in &keys {
224            test_exists.push(store.contains_key(key).await.unwrap());
225            values_single_read.push(store.read_value_bytes(key).await.unwrap());
226        }
227        let test_exists_direct = store.contains_keys(keys.clone()).await.unwrap();
228        let values_read = store.read_multi_values_bytes(keys).await.unwrap();
229        assert_eq!(values, values_read);
230        assert_eq!(values, values_single_read);
231        let values_read_stat = values_read.iter().map(|x| x.is_some()).collect::<Vec<_>>();
232        assert_eq!(values_read_stat, test_exists);
233        assert_eq!(values_read_stat, test_exists_direct);
234    }
235}

243/// Generates a list of random key-values with no duplicates
244pub fn get_random_key_values2(
245    num_entries: usize,
246    len_key: usize,
247    len_value: usize,
248) -> Vec<(Vec<u8>, Vec<u8>)> {
249    let mut rng = make_deterministic_rng();
250    let key_prefix = vec![0];
251    let mut key_values = Vec::new();
252    let mut key_set = HashSet::new();
253    for _ in 0..num_entries {
254        let key = get_small_key_space(&mut rng, &key_prefix, len_key);
255        if !key_set.contains(&key) {
256            key_set.insert(key.clone());
257            let value = get_random_byte_vector(&mut rng, &[], len_value);
258            key_values.push((key, value));
259        }
260    }
261    key_values
262}

287fn generate_random_batch<R: Rng>(rng: &mut R, key_prefix: &[u8], batch_size: usize) -> Batch {
288    let mut batch = Batch::new();
289    // Fully random batch
290    for _ in 0..batch_size {
291        let choice = rng.gen_range(0..8);
292        // Inserting a key
293        if choice < 6 {
294            // Insert
295            let key = get_small_key_space(rng, key_prefix, 4);
296            let len_value = rng.gen_range(0..10); // Could need to be split
297            let value = get_random_byte_vector(rng, &[], len_value);
298            batch.put_key_value_bytes(key.clone(), value.clone());
299        }
300        if choice == 6 {
301            // key might be missing, no matter, it has to work
302            let key = get_small_key_space(rng, key_prefix, 4);
303            batch.delete_key(key);
304        }
305        if choice == 7 {
306            let len = rng.gen_range(1..4); // We want a non-trivial range
307            let delete_key_prefix = get_small_key_space(rng, key_prefix, len);
308            batch.delete_key_prefix(delete_key_prefix.clone());
309        }
310    }
311    batch
312}

320fn generate_specific_batch(key_prefix: &[u8], option: usize) -> Batch {
321    let mut batch = Batch::new();
322    if option == 0 {
323        let key = get_key(key_prefix, vec![34]);
324        batch.put_key_value_bytes(key.clone(), vec![]);
325        batch.delete_key(key);
326    }
327    if option == 1 {
328        let key1 = get_key(key_prefix, vec![12, 34]);
329        let key2 = get_key(key_prefix, vec![12, 33]);
330        let key3 = get_key(key_prefix, vec![13]);
331        batch.put_key_value_bytes(key1.clone(), vec![]);
332        batch.put_key_value_bytes(key2, vec![]);
333        batch.put_key_value_bytes(key3, vec![]);
334        batch.delete_key(key1);
335        let key_prefix = get_key(key_prefix, vec![12]);
336        batch.delete_key_prefix(key_prefix);
337    }
338    batch
339}

341fn update_state_from_batch(kv_state: &mut BTreeMap<Vec<u8>, Vec<u8>>, batch: &Batch) {
342    for operation in &batch.operations {
343        match operation {
344            WriteOperation::Put { key, value } => {
345                kv_state.insert(key.to_vec(), value.to_vec());
346            }
347            WriteOperation::Delete { key } => {
348                kv_state.remove(key);
349            }
350            WriteOperation::DeletePrefix { key_prefix } => {
351                kv_state.retain(|key, _| !key.starts_with(key_prefix));
352            }
353        }
354    }
355}

357fn realize_batch(batch: &Batch) -> BTreeMap<Vec<u8>, Vec<u8>> {
358    let mut kv_state = BTreeMap::new();
359    update_state_from_batch(&mut kv_state, batch);
360    kv_state
361}

363async fn read_keys_prefix<C: RestrictedKeyValueStore>(
364    key_value_store: &C,
365    key_prefix: &[u8],
366) -> BTreeSet<Vec<u8>> {
367    let mut keys = BTreeSet::new();
368    for key in key_value_store
369        .find_keys_by_prefix(key_prefix)
370        .await
371        .unwrap()
372        .iterator()
373    {
374        let key_suffix = key.unwrap();
375        let mut key = key_prefix.to_vec();
376        key.extend(key_suffix);
377        keys.insert(key);
378    }
379    keys
380}

382async fn read_key_values_prefix<C: RestrictedKeyValueStore>(
383    key_value_store: &C,
384    key_prefix: &[u8],
385) -> BTreeMap<Vec<u8>, Vec<u8>> {
386    let mut key_values = BTreeMap::new();
387    for key_value in key_value_store

389        .await
390        .unwrap()
391        .iterator()
392    {
393        let (key_suffix, value) = key_value.unwrap();
394        let mut key = key_prefix.to_vec();
395        key.extend(key_suffix);
396        key_values.insert(key, value.to_vec());
397    }
398    key_values
399}

401/// Writes and then reads data under a prefix, and verifies the result.
402pub async fn run_test_batch_from_blank<C: RestrictedKeyValueStore>(
403    key_value_store: &C,
404    key_prefix: Vec<u8>,
405    batch: Batch,
406) {
407    let kv_state = realize_batch(&batch);
408    key_value_store.write_batch(batch).await.unwrap();
409    // Checking the consistency
410    let key_values = read_key_values_prefix(key_value_store, &key_prefix).await;
411    assert_eq!(key_values, kv_state);
412}

414/// Run many operations on batches always starting from a blank state.
415pub async fn run_writes_from_blank<C: RestrictedKeyValueStore>(key_value_store: &C) {
416    let mut rng = make_deterministic_rng();
417    let n_oper = 10;
418    let batch_size = 500;
419    // key space has size 4^4 = 256 so we necessarily encounter collisions
420    // because the number of generated keys is about batch_size * n_oper = 800 > 256.
421    for _ in 0..n_oper {
422        let key_prefix = get_random_key_prefix();
423        let batch = generate_random_batch(&mut rng, &key_prefix, batch_size);
424        run_test_batch_from_blank(key_value_store, key_prefix, batch).await;
425    }
426    for option in 0..2 {
427        let key_prefix = get_random_key_prefix();
428        let batch = generate_specific_batch(&key_prefix, option);
429        run_test_batch_from_blank(key_value_store, key_prefix, batch).await;
430    }
431}

433/// Reading many keys at a time could trigger an error. This needs to be tested.
434pub async fn big_read_multi_values<C: KeyValueStore>(
435    config: C::Config,
436    value_size: usize,
437    n_entries: usize,
438) {
439    let mut rng = make_deterministic_rng();
440    let namespace = generate_test_namespace();
441    let store = C::recreate_and_connect(&config, &namespace).await.unwrap();
442    let store = store.clone_with_root_key(&[]).unwrap();
443    let key_prefix = vec![42, 54];
444    let mut batch = Batch::new();
445    let mut keys = Vec::new();
446    let mut values = Vec::new();
447    for i in 0..n_entries {
448        let mut key = key_prefix.clone();
449        bcs::serialize_into(&mut key, &i).unwrap();
450        let value = get_random_byte_vector(&mut rng, &[], value_size);

452        keys.push(key);
453        values.push(Some(value));

456    // We reconnect so that the read is not using the cache.
457    let store = C::connect(&config, &namespace).await.unwrap();
458    let store = store.clone_with_root_key(&[]).unwrap();
459    let values_read = store.read_multi_values_bytes(keys).await.unwrap();
460    assert_eq!(values, values_read);
461}

463/// That test is especially challenging for ScyllaDB.
464/// In its default settings, Scylla has a limitation to 10000 tombstones.
465/// A tombstone is an indication that the data has been deleted. That
466/// is thus a trie data structure for checking whether a requested key
467/// is deleted or not.
468///
469/// In this test we insert 200000 keys into the database.
470/// Then we select half of them at random and delete them. By the random
471/// selection, Scylla is forced to introduce around 100000 tombstones
472/// which triggers the crash with the default settings.
473pub async fn tombstone_triggering_test<C: RestrictedKeyValueStore>(key_value_store: C) {
474    use std::time::Instant;
475    let t1 = Instant::now();
476    let mut rng = make_deterministic_rng();
477    let value_size = 100;
478    let n_entry = 200000;
479    // Putting the keys
480    let mut batch_insert = Batch::new();
481    let key_prefix = vec![0];
482    let mut batch_delete = Batch::new();
483    let mut remaining_key_values = BTreeMap::new();
484    let mut remaining_keys = BTreeSet::new();
485    for i in 0..n_entry {
486        let mut key = key_prefix.clone();
487        bcs::serialize_into(&mut key, &i).unwrap();
488        let value = get_random_byte_vector(&mut rng, &[], value_size);
489        batch_insert.put_key_value_bytes(key.clone(), value.clone());
490        let to_delete = rng.gen::<bool>();
491        if to_delete {
492            batch_delete.delete_key(key);
493        } else {
494            remaining_keys.insert(key.clone());
495            remaining_key_values.insert(key, value);
496        }
497    }
498    tracing::info!("Set up in {} ms", t1.elapsed().as_millis());
499
500    let t1 = Instant::now();
501    run_test_batch_from_blank(&key_value_store, key_prefix.clone(), batch_insert).await;
502    tracing::info!("run_test_batch in {} ms", t1.elapsed().as_millis());
503
504    // Deleting them all
505    let t1 = Instant::now();
506    key_value_store.write_batch(batch_delete).await.unwrap();
507    tracing::info!("batch_delete in {} ms", t1.elapsed().as_millis());
508
509    for iter in 0..5 {
510        // Reading everything and seeing that it is now cleaned.
511        let t1 = Instant::now();
512        let key_values = read_key_values_prefix(&key_value_store, &key_prefix).await;
513        assert_eq!(key_values, remaining_key_values);
514        tracing::info!(
515            "iter={} read_key_values_prefix in {} ms",
516            iter,
517            t1.elapsed().as_millis()
518        );
519
520        let t1 = Instant::now();
521        let keys = read_keys_prefix(&key_value_store, &key_prefix).await;
522        assert_eq!(keys, remaining_keys);
523        tracing::info!(
524            "iter={} read_keys_prefix after {} ms",
525            iter,
526            t1.elapsed().as_millis()
527        );
528    }
529}

531/// DynamoDB has limits at 1 MB (for pagination), 4 MB (for write)
532/// Let us go right past them at 20 MB of data with writing and then
533/// reading it. And 20 MB is not huge by any mean. All `KeyValueStore`
534/// must handle that.
535///
536/// The size of the value vary as each size has its own issues.
537pub async fn run_big_write_read<C: RestrictedKeyValueStore>(
538    key_value_store: C,
539    target_size: usize,
540    value_sizes: Vec<usize>,
541) {
542    let mut rng = make_deterministic_rng();
543    for (pos, value_size) in value_sizes.into_iter().enumerate() {
544        let n_entry: usize = target_size / value_size;
545        let mut batch = Batch::new();
546        let key_prefix = vec![0, pos as u8];
547        for i in 0..n_entry {
548            let mut key = key_prefix.clone();
549            bcs::serialize_into(&mut key, &i).unwrap();
550            let value = get_random_byte_vector(&mut rng, &[], value_size);
551            batch.put_key_value_bytes(key, value);
552        }
553        run_test_batch_from_blank(&key_value_store, key_prefix, batch).await;
554    }
555}

557type StateBatch = (Vec<(Vec<u8>, Vec<u8>)>, Batch);
558
559async fn run_test_batch_from_state<C: RestrictedKeyValueStore>(
560    key_value_store: &C,
561    key_prefix: Vec<u8>,
562    state_and_batch: StateBatch,
563) {
564    let (key_values, batch) = state_and_batch;
565    let mut batch_insert = Batch::new();
566    let mut kv_state = BTreeMap::new();
567    for (key, value) in key_values {
568        kv_state.insert(key.clone(), value.clone());
569        batch_insert.put_key_value_bytes(key, value);
570    }
571    key_value_store.write_batch(batch_insert).await.unwrap();
572    let key_values = read_key_values_prefix(key_value_store, &key_prefix).await;
573    assert_eq!(key_values, kv_state);
574
575    update_state_from_batch(&mut kv_state, &batch);
576    key_value_store.write_batch(batch).await.unwrap();
577    let key_values = read_key_values_prefix(key_value_store, &key_prefix).await;
578    assert_eq!(key_values, kv_state);
579}

581fn generate_specific_state_batch(key_prefix: &[u8], option: usize) -> StateBatch {
582    let mut key_values = Vec::new();
583    let mut batch = Batch::new();
584    if option == 0 {
585        // A DeletePrefix followed by an insertion that matches the DeletePrefix
586        let key1 = get_key(key_prefix, vec![1, 3]);
587        let key2 = get_key(key_prefix, vec![1, 4]);
588        let key3 = get_key(key_prefix, vec![1, 4, 5]);
589        key_values.push((key1.clone(), vec![34]));
590        key_values.push((key2.clone(), vec![45]));
591        batch.delete_key_prefix(key2);
592        batch.put_key_value_bytes(key3, vec![23]);
593    }
594    if option == 1 {
595        // Just a DeletePrefix
596        let key1 = get_key(key_prefix, vec![1, 3]);
597        let key2 = get_key(key_prefix, vec![1, 4]);
598        key_values.push((key1.clone(), vec![34]));
599        key_values.push((key2.clone(), vec![45]));
600        batch.delete_key_prefix(key2);
601    }
602    if option == 2 {
603        // A Put followed by a DeletePrefix that matches the Put
604        let key1 = get_key(key_prefix, vec![1, 3]);
605        let key2 = get_key(key_prefix, vec![1, 4]);
606        let key3 = get_key(key_prefix, vec![1, 4, 5]);
607        key_values.push((key1.clone(), vec![34]));
608        key_values.push((key2.clone(), vec![45]));
609        batch.put_key_value_bytes(key3, vec![23]);
610        batch.delete_key_prefix(key2);
611    }
612    if option == 3 {
613        // A Put followed by a Delete on the same value
614        let key1 = get_key(key_prefix, vec![1, 3]);
615        let key2 = get_key(key_prefix, vec![1, 4]);
616        let key3 = get_key(key_prefix, vec![1, 4, 5]);
617        key_values.push((key1.clone(), vec![34]));
618        key_values.push((key2.clone(), vec![45]));
619        batch.put_key_value_bytes(key3.clone(), vec![23]);
620        batch.delete_key(key3);
621    }
622    if option == 4 {
623        // A Delete Key followed by a Put on the same key
624        let key1 = get_key(key_prefix, vec![1, 3]);
625        let key2 = get_key(key_prefix, vec![1, 4]);
626        let key3 = get_key(key_prefix, vec![1, 4, 5]);
627        key_values.push((key1.clone(), vec![34]));
628        key_values.push((key2.clone(), vec![45]));
629        batch.delete_key(key3.clone());
630        batch.put_key_value_bytes(key3, vec![23]);
631    }
632    if option == 5 {
633        // A Delete Key followed by a Put on the same key
634        let key1 = get_key(key_prefix, vec![1, 3]);
635        let key2 = get_key(key_prefix, vec![1, 4]);
636        let key3 = get_key(key_prefix, vec![1, 4, 5]);
637        let key4 = get_key(key_prefix, vec![1, 5]);
638        key_values.push((key1.clone(), vec![34]));
639        key_values.push((key2.clone(), vec![45]));
640        batch.delete_key(key3.clone());
641        batch.put_key_value_bytes(key4, vec![23]);
642    }
643    if option == 6 {
644        let key1 = get_key(key_prefix, vec![0]);
645        let key2 = get_key(key_prefix, vec![]);
646        key_values.push((key1, vec![33]));
647        batch.delete_key_prefix(key2);
648    }
649    if option == 7 {
650        let key1 = get_key(key_prefix, vec![255, 255]);
651        let key2 = get_key(key_prefix, vec![255, 255, 1]);
652        key_values.push((key2.clone(), vec![]));
653        batch.delete_key_prefix(key1);
654        batch.put_key_value_bytes(key2, vec![]);
655    }
656    (key_values, batch)
657}

659/// Run some deterministic and random batches operation and check their
660/// correctness
661pub async fn run_writes_from_state<C: RestrictedKeyValueStore>(key_value_store: &C) {
662    for option in 0..8 {
663        let key_prefix = if option >= 6 {
664            vec![255, 255, 255]
665        } else {
666            get_random_key_prefix()
667        };
668        let state_batch = generate_specific_state_batch(&key_prefix, option);
669        run_test_batch_from_state(key_value_store, key_prefix, state_batch).await;
670    }

673async fn namespaces_with_prefix<S: KeyValueStore>(
674    config: &S::Config,
675    prefix: &str,

677    let namespaces = S::list_all(config).await.expect("namespaces");

684/// Exercises the namespace functionalities of the `AdminKeyValueStore`.
685/// This tests everything except the `delete_all` which would
686/// interact with other namespaces.
687pub async fn namespace_admin_test<S: TestKeyValueStore>() {
688    let config = S::new_test_config().await.expect("config");
689    {
690        let namespace = generate_test_namespace();
691        S::create(&config, &namespace)
692            .await
693            .expect("first creation of a namespace");
694        // Creating a namespace two times should returns an error
695        assert!(S::create(&config, &namespace).await.is_err());
696    }
697    let prefix = generate_test_namespace();
698    let namespaces = namespaces_with_prefix::<S>(&config, &prefix).await;
699    assert_eq!(namespaces.len(), 0);
700    let mut rng = make_deterministic_rng();
701    let size = 9;
702    // Creating the initial list of namespaces
703    let mut working_namespaces = BTreeSet::new();
704    for i in 0..size {
705        let namespace = format!("{}_{}", prefix, i);
706        assert!(!S::exists(&config, &namespace).await.expect("test"));
707        working_namespaces.insert(namespace);
708    }
709    // Creating the namespaces
710    for namespace in &working_namespaces {
711        S::create(&config, namespace)
712            .await
713            .expect("creation of a namespace");
714        assert!(S::exists(&config, namespace).await.expect("test"));
715    }
716    // Connecting to all of them at once
717    {
718        let mut connections = Vec::new();
719        for namespace in &working_namespaces {
720            let connection = S::connect(&config, namespace)
721                .await
722                .expect("a connection to the namespace");
723            connections.push(connection);
724        }
725    }
726    // Listing all of them
727    let namespaces = namespaces_with_prefix::<S>(&config, &prefix).await;
728    assert_eq!(namespaces, working_namespaces);
729    // Selecting at random some for deletion
730    let mut kept_namespaces = BTreeSet::new();
731    for namespace in working_namespaces {
732        let delete = rng.gen::<bool>();
733        if delete {
734            S::delete(&config, &namespace)
735                .await
736                .expect("A successful deletion");
737            assert!(!S::exists(&config, &namespace).await.expect("test"));
738        } else {
739            kept_namespaces.insert(namespace);
740        }
741    }
742    for namespace in &kept_namespaces {
743        assert!(S::exists(&config, namespace).await.expect("test"));
744    }
745    let namespaces = namespaces_with_prefix::<S>(&config, &prefix).await;
746    assert_eq!(namespaces, kept_namespaces);
747    for namespace in kept_namespaces {
748        S::delete(&config, &namespace)
749            .await
750            .expect("A successful deletion");
751    }
752}

754/// Tests listing the root keys.
755pub async fn root_key_admin_test<S: TestKeyValueStore>() {
756    let config = S::new_test_config().await.expect("config");
757    let namespace = generate_test_namespace();
758    let mut root_keys = Vec::new();
759    let mut keys = BTreeSet::new();
760    S::create(&config, &namespace).await.expect("creation");
761    let prefix = vec![0];
762    {
763        let size = 3;
764        let mut rng = make_deterministic_rng();
765        let store = S::connect(&config, &namespace).await.expect("store");
766        root_keys.push(vec![]);
767        let mut batch = Batch::new();
768        for _ in 0..2 {
769            let key = get_random_byte_vector(&mut rng, &prefix, 4);
770            batch.put_key_value_bytes(key.clone(), vec![]);
771            keys.insert((vec![], key));
772        }
773        store.write_batch(batch).await.expect("write batch");
774
775        for _ in 0..20 {
776            let root_key = get_random_byte_vector(&mut rng, &[], 4);
777            let cloned_store = store.clone_with_root_key(&root_key).expect("cloned store");
778            root_keys.push(root_key.clone());
779            let size_select = rng.gen_range(0..size);
780            let mut batch = Batch::new();
781            for _ in 0..size_select {
782                let key = get_random_byte_vector(&mut rng, &prefix, 4);
783                batch.put_key_value_bytes(key.clone(), vec![]);
784                keys.insert((root_key.clone(), key));
785            }
786            cloned_store.write_batch(batch).await.expect("write batch");
787        }
788    }
789
790    let read_root_keys = S::list_root_keys(&config, &namespace)
791        .await
792        .expect("read_root_keys");
793    let set_root_keys = root_keys.iter().cloned().collect::<HashSet<_>>();
794    for read_root_key in &read_root_keys {
795        assert!(set_root_keys.contains(read_root_key));
796    }
797
798    let mut read_keys = BTreeSet::new();
799    for root_key in read_root_keys {
800        let store = S::connect(&config, &namespace)
801            .await
802            .expect("store")
803            .clone_with_root_key(&root_key)
804            .expect("clone_with_root_key");
805        let keys = store.find_keys_by_prefix(&prefix).await.expect("keys");
806        for key in keys.iterator() {
807            let key = key.expect("key");
808            let mut big_key = prefix.clone();
809            let key = key.to_vec();
810            big_key.extend(key);
811            read_keys.insert((root_key.clone(), big_key));
812        }
813    }
814    assert_eq!(keys, read_keys);
815}

817/// A store can be in exclusive access where it stores the absence of values
818/// or in shared access where only values are stored and (key, value) once
819/// written are never modified nor erased.
820///
821/// In case of no exclusive access the following scenario is checked
822/// * Store 1 deletes a key and does not mark it as missing in its cache.
823/// * Store 2 writes the key
824/// * Store 1 reads the key, but since it is not in the cache it can read
825///   it correctly.
826///
827/// In case of exclusive access. We have the following scenario:
828/// * Store 1 deletes a key and mark it as missing in its cache.
829/// * Store 2 writes the key (it should not be doing it)
830/// * Store 1 reads the key, see it as missing.
831pub async fn exclusive_access_admin_test<S: TestKeyValueStore>(exclusive_access: bool) {
832    let config = S::new_test_config().await.expect("config");
833    let namespace = generate_test_namespace();
834    S::create(&config, &namespace).await.expect("creation");
835    let key = vec![42];
836
837    let mut store1 = S::connect(&config, &namespace).await.expect("store");
838    if exclusive_access {
839        store1 = store1.clone_with_root_key(&[]).expect("store1");
840    }
841    let mut batch1 = Batch::new();
842    batch1.delete_key(key.clone());
843    store1.write_batch(batch1).await.expect("write batch1");
844
845    let mut store2 = S::connect(&config, &namespace).await.expect("store");
846    if exclusive_access {
847        store2 = store2.clone_with_root_key(&[]).expect("store2");
848    }
849    let mut batch2 = Batch::new();
850    batch2.put_key_value_bytes(key.clone(), vec![]);
851    store2.write_batch(batch2).await.expect("write batch2");
852
853    assert_eq!(store1.contains_key(&key).await.unwrap(), !exclusive_access);
854}
