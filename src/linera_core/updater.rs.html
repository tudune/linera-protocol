35/// The default amount of time we wait for additional validators to contribute
36/// to the result, as a fraction of how long it took to reach a quorum.
37pub const DEFAULT_GRACE_PERIOD: f64 = 0.2;
38/// The maximum timeout for requests to a stake-weighted quorum if no quorum is reached.
39const MAX_TIMEOUT: Duration = Duration::from_secs(60 * 60 * 24); // 1 day.
40
41/// Used for `communicate_chain_action`
42#[derive(Clone)]
43pub enum CommunicateAction {
44    SubmitBlock {
45        proposal: Box<BlockProposal>,
46        blob_ids: Vec<BlobId>,
47    },
48    FinalizeBlock {
49        certificate: Box<ValidatedBlockCertificate>,
50        delivery: CrossChainMessageDelivery,
51    },
52    RequestTimeout {
53        chain_id: ChainId,
54        height: BlockHeight,
55        round: Round,
56    },
57}
58
59impl CommunicateAction {
60    /// The round to which this action pertains.
61    pub fn round(&self) -> Round {
62        match self {
63            CommunicateAction::SubmitBlock { proposal, .. } => proposal.content.round,
64            CommunicateAction::FinalizeBlock { certificate, .. } => certificate.round,
65            CommunicateAction::RequestTimeout { round, .. } => *round,
66        }
67    }
68}

80/// An error result for requests to a stake-weighted quorum.
81#[derive(Error, Debug)]
82pub enum CommunicationError<E: fmt::Debug> {
83    /// No consensus is possible since validators returned different possibilities
84    /// for the next block
85    #[error(
86        "No error but failed to find a consensus block. Consensus threshold: {0}, Proposals: {1:?}"
87    )]
88    NoConsensus(u64, Vec<(u64, usize)>),
89    /// A single error that was returned by a sufficient number of nodes to be trusted as
90    /// valid.
91    #[error("Failed to communicate with a quorum of validators: {0}")]
92    Trusted(E),
93    /// No single error reached the validity threshold so we're returning a sample of
94    /// errors for debugging purposes, together with their weight.
95    #[error("Failed to communicate with a quorum of validators:\n{:#?}", .0)]
96    Sample(Vec<(E, u64)>),
97}
98
99/// Executes a sequence of actions in parallel for all validators.
100///
101/// Tries to stop early when a quorum is reached. If `grace_period` is specified, other validators
102/// are given additional time to contribute to the result. The grace period is calculated as a fraction
103/// (defaulting to `DEFAULT_GRACE_PERIOD`) of the time taken to reach quorum.
104pub async fn communicate_with_quorum<'a, A, V, K, F, R, G>(
105    validator_clients: &'a [RemoteNode<A>],
106    committee: &Committee,
107    group_by: G,
108    execute: F,
109    // Grace period as a fraction of time taken to reach quorum
110    grace_period: f64,
111) -> Result<(K, Vec<V>), CommunicationError<NodeError>>
112where
113    A: ValidatorNode + Clone + 'static,
114    F: Clone + Fn(RemoteNode<A>) -> R,
115    R: Future<Output = Result<V, ChainClientError>> + 'a,
116    G: Fn(&V) -> K,
117    K: Hash + PartialEq + Eq + Clone + 'static,
118    V: 'static,
119{
120    let mut responses: futures::stream::FuturesUnordered<_> = validator_clients
121        .iter()
122        .filter_map(|remote_node| {
123            if committee.weight(&remote_node.public_key) == 0 {
124                // This should not happen but better prevent it because certificates
125                // are not allowed to include votes with weight 0.
126                return None;
127            }
128            let execute = execute.clone();
129            let remote_node = remote_node.clone();
130            Some(async move { (remote_node.public_key, execute(remote_node).await) })
131        })
132        .collect();
133
134    let start_time = Instant::now();
135    let mut end_time: Option<Instant> = None;
136    let mut remaining_votes = committee.total_votes();
137    let mut highest_key_score = 0;
138    let mut value_scores = HashMap::new();
139    let mut error_scores = HashMap::new();
140
141    'vote_wait: while let Ok(Some((name, result))) = timeout(
142        end_time.map_or(MAX_TIMEOUT, |t| t.saturating_duration_since(Instant::now())),
143        responses.next(),
144    )
145    .await
146    {
147        remaining_votes -= committee.weight(&name);
148        match result {
149            Ok(value) => {
150                let key = group_by(&value);
151                let entry = value_scores.entry(key.clone()).or_insert((0, Vec::new()));
152                entry.0 += committee.weight(&name);
153                entry.1.push(value);
154                highest_key_score = highest_key_score.max(entry.0);
155            }
156            Err(err) => {
157                // TODO(#2857): Handle non-remote errors properly.
158                let err = match err {
159                    ChainClientError::RemoteNodeError(err) => err,
160                    err => NodeError::ResponseHandlingError {
161                        error: err.to_string(),
162                    },
163                };
164                let entry = error_scores.entry(err.clone()).or_insert(0);
165                *entry += committee.weight(&name);
166                if *entry >= committee.validity_threshold() {
167                    // At least one honest node returned this error.
168                    // No quorum can be reached, so return early.
169                    return Err(CommunicationError::Trusted(err));
170                }
171            }
172        }
173        // If it becomes clear that no key can reach a quorum, break early.
174        if highest_key_score + remaining_votes < committee.quorum_threshold() {
175            break 'vote_wait;
176        }
177
178        // If a key reaches a quorum, wait for the grace period to collect more values
179        // or error information and then stop.
180        if end_time.is_none() && highest_key_score >= committee.quorum_threshold() {
181            end_time = Some(Instant::now() + start_time.elapsed().mul_f64(grace_period));
182        }
183    }
184
185    let scores = value_scores
186        .values()
187        .map(|(weight, values)| (*weight, values.len()))
188        .collect();
189    // If a key has a quorum, return it with its values.
190    if let Some((key, (_, values))) = value_scores
191        .into_iter()
192        .find(|(_, (score, _))| *score >= committee.quorum_threshold())
193    {
194        return Ok((key, values));
195    }
196
197    if error_scores.is_empty() {
198        return Err(CommunicationError::NoConsensus(
199            committee.quorum_threshold(),
200            scores,
201        ));
202    }
203
204    // No specific error is available to report reliably.
205    let mut sample = error_scores.into_iter().collect::<Vec<_>>();
206    sample.sort_by_key(|(_, score)| std::cmp::Reverse(*score));
207    sample.truncate(4);
208    Err(CommunicationError::Sample(sample))
209}

216    async fn send_confirmed_certificate(
217        &mut self,
218        certificate: GenericCertificate<ConfirmedBlock>,
219        delivery: CrossChainMessageDelivery,
220    ) -> Result<Box<ChainInfo>, ChainClientError> {
221        let result = self
222            .remote_node
223            .handle_optimized_confirmed_certificate(&certificate, delivery)
224            .await;
225
226        Ok(match &result {
227            Err(original_err @ NodeError::BlobsNotFound(blob_ids)) => {
228                self.remote_node
229                    .check_blobs_not_found(&certificate, blob_ids)?;
230                // The certificate is confirmed, so the blobs must be in storage.
231                let maybe_blobs = self.local_node.read_blobs_from_storage(blob_ids).await?;
232                let blobs = maybe_blobs.ok_or_else(|| original_err.clone())?;
233                self.remote_node.upload_blobs(blobs.clone()).await?;
234                self.remote_node
235                    .handle_confirmed_certificate(certificate, delivery)
236                    .await
237            }
238            _ => result,
239        }?)
240    }
241
242    async fn send_validated_certificate(
243        &mut self,
244        certificate: GenericCertificate<ValidatedBlock>,
245        delivery: CrossChainMessageDelivery,
246    ) -> Result<Box<ChainInfo>, ChainClientError> {
247        let result = self
248            .remote_node
249            .handle_optimized_validated_certificate(&certificate, delivery)
250            .await;
251
252        Ok(match &result {
253            Err(original_err @ NodeError::BlobsNotFound(blob_ids)) => {
254                self.remote_node
255                    .check_blobs_not_found(&certificate, blob_ids)?;
256                let chain_id = certificate.inner().chain_id();
257                // The certificate is for a validated block, i.e. for our locking block.
258                // Take the missing blobs from our local chain manager.
259                let blobs = self
260                    .local_node
261                    .get_locking_blobs(blob_ids, chain_id)
262                    .await?
263                    .ok_or_else(|| original_err.clone())?;
264                self.remote_node.send_pending_blobs(chain_id, blobs).await?;
265                self.remote_node
266                    .handle_validated_certificate(certificate)
267                    .await
268            }
269            _ => result,
270        }?)
271    }

273    async fn send_block_proposal(
274        &mut self,
275        proposal: Box<BlockProposal>,
276        mut blob_ids: Vec<BlobId>,
277    ) -> Result<Box<ChainInfo>, ChainClientError> {
278        let chain_id = proposal.content.block.chain_id;
279        let mut sent_cross_chain_updates = false;
280        loop {
281            match self
282                .remote_node
283                .handle_block_proposal(proposal.clone())
284                .await
285            {
286                Ok(info) => return Ok(info),
287                Err(NodeError::MissingCrossChainUpdate { .. })
288                | Err(NodeError::InactiveChain(_))
289                    if !sent_cross_chain_updates =>
290                {
291                    sent_cross_chain_updates = true;
292                    // Some received certificates may be missing for this validator
293                    // (e.g. to create the chain or make the balance sufficient) so we are going to
294                    // synchronize them now and retry.
295                    self.send_chain_information_for_senders(chain_id).await?;
296                }
297                Err(NodeError::BlobsNotFound(_)) if !blob_ids.is_empty() => {
298                    // For `BlobsNotFound`, we assume that the local node should already be
299                    // updated with the needed blobs, so sending the chain information about the
300                    // certificates that last used the blobs to the validator node should be enough.
301                    let published_blob_ids =
302                        BTreeSet::from_iter(proposal.content.block.published_blob_ids());
303                    blob_ids.retain(|blob_id| !published_blob_ids.contains(blob_id));
304                    let mut published_blobs = Vec::new();
305                    {
306                        let chain = self.local_node.chain_state_view(chain_id).await?;
307                        for blob_id in published_blob_ids {
308                            published_blobs
309                                .extend(chain.manager.proposed_blobs.get(&blob_id).await?);
310                        }
311                    }
312                    self.remote_node
313                        .send_pending_blobs(chain_id, published_blobs)
314                        .await?;
315                    let missing_blob_ids = self
316                        .remote_node
317                        .node
318                        .missing_blob_ids(mem::take(&mut blob_ids))
319                        .await?;
320                    let blob_states = self
321                        .local_node
322                        .read_blob_states_from_storage(&missing_blob_ids)
323                        .await?;
324                    let mut chain_heights = BTreeMap::new();
325                    for blob_state in blob_states {
326                        let block_chain_id = blob_state.chain_id;
327                        let block_height = blob_state.block_height.try_add_one()?;
328                        chain_heights
329                            .entry(block_chain_id)
330                            .and_modify(|h| *h = block_height.max(*h))
331                            .or_insert(block_height);
332                    }
333
334                    self.send_chain_info_up_to_heights(
335                        chain_heights,
336                        CrossChainMessageDelivery::NonBlocking,
337                    )
338                    .await?;
339                }
340                // Fail immediately on other errors.
341                Err(err) => return Err(err.into()),
342            }
343        }
344    }

346    pub async fn send_chain_information(
347        &mut self,
348        chain_id: ChainId,
349        target_block_height: BlockHeight,
350        delivery: CrossChainMessageDelivery,
351    ) -> Result<(), ChainClientError> {
352        // Figure out which certificates this validator is missing.
353        let query = ChainInfoQuery::new(chain_id);
354        let remote_info = self.remote_node.handle_chain_info_query(query).await?;
355        let initial_block_height = remote_info.next_block_height;
356        // Obtain the missing blocks and the manager state from the local node.
357        let range: Range<usize> =
358            initial_block_height.try_into()?..target_block_height.try_into()?;
359        let (keys, timeout) = {
360            let chain = self.local_node.chain_state_view(chain_id).await?;
361            (
362                chain.confirmed_log.read(range).await?,
363                chain.manager.timeout.get().clone(),
364            )
365        };
366        if !keys.is_empty() {
367            // Send the requested certificates in order.
368            let storage = self.local_node.storage_client();
369            let certs = storage.read_certificates(keys.into_iter()).await?;
370            for cert in certs {
371                self.send_confirmed_certificate(cert, delivery).await?;
372            }
373        }
374        if let Some(cert) = timeout {
375            if cert.value().chain_id() == chain_id {
376                // Timeouts are small and don't have blobs, so we can call `handle_certificate`
377                // directly.
378                self.remote_node.handle_timeout_certificate(cert).await?;
379            }
380        }
381        Ok(())
382    }
383
384    async fn send_chain_info_up_to_heights(
385        &mut self,
386        chain_heights: BTreeMap<ChainId, BlockHeight>,
387        delivery: CrossChainMessageDelivery,
388    ) -> Result<(), ChainClientError> {
389        let stream = stream::iter(chain_heights)
390            .map(|(chain_id, height)| {
391                let mut updater = self.clone();
392                async move {
393                    updater
394                        .send_chain_information(chain_id, height, delivery)
395                        .await
396                }
397            })
398            .buffer_unordered(self.chain_worker_count);
399        stream.try_collect::<Vec<_>>().await?;
400        Ok(())
401    }
402
403    async fn send_chain_information_for_senders(
404        &mut self,
405        chain_id: ChainId,
406    ) -> Result<(), ChainClientError> {
407        let mut sender_heights = BTreeMap::new();
408        {
409            let chain = self.local_node.chain_state_view(chain_id).await?;
410            let pairs = chain.inboxes.try_load_all_entries().await?;
411            for (origin, inbox) in pairs {
412                let inbox_next_height = inbox.next_block_height_to_receive()?;
413                sender_heights
414                    .entry(origin)
415                    .and_modify(|h| *h = inbox_next_height.max(*h))
416                    .or_insert(inbox_next_height);
417            }
418        }
419
420        self.send_chain_info_up_to_heights(sender_heights, CrossChainMessageDelivery::Blocking)
421            .await?;
422        Ok(())
423    }

425    pub async fn send_chain_update(
426        &mut self,
427        action: CommunicateAction,
428    ) -> Result<LiteVote, ChainClientError> {
429        let (target_block_height, chain_id) = match &action {
430            CommunicateAction::SubmitBlock { proposal, .. } => {
431                let block = &proposal.content.block;
432                (block.height, block.chain_id)
433            }
434            CommunicateAction::FinalizeBlock { certificate, .. } => (
435                certificate.inner().block().header.height,
436                certificate.inner().block().header.chain_id,
437            ),
438            CommunicateAction::RequestTimeout {
439                height, chain_id, ..
440            } => (*height, *chain_id),
441        };
442        // Update the validator with missing information, if needed.
443        let delivery = CrossChainMessageDelivery::NonBlocking;
444        self.send_chain_information(chain_id, target_block_height, delivery)
445            .await?;
446        // Send the block proposal, certificate or timeout request and return a vote.
447        let vote = match action {
448            CommunicateAction::SubmitBlock { proposal, blob_ids } => {
449                let info = self.send_block_proposal(proposal, blob_ids).await?;
450                info.manager.pending
451            }
452            CommunicateAction::FinalizeBlock {
453                certificate,
454                delivery,
455            } => {
456                let info = self
457                    .send_validated_certificate(*certificate, delivery)
458                    .await?;
459                info.manager.pending
460            }
461            CommunicateAction::RequestTimeout { .. } => {
462                let query = ChainInfoQuery::new(chain_id).with_timeout();
463                let info = self.remote_node.handle_chain_info_query(query).await?;
464                info.manager.timeout_vote
465            }
466        };
467        match vote {
468            Some(vote) if vote.public_key == self.remote_node.public_key => {
469                vote.check()?;
470                Ok(vote)
471            }
472            Some(_) | None => Err(NodeError::MissingVoteInValidatorResponse.into()),
473        }
474    }
