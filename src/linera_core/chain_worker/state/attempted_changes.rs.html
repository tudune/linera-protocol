38/// Wrapper type that tracks if the changes to the `chain` state should be rolled back when
39/// dropped.
40pub struct ChainWorkerStateWithAttemptedChanges<'state, StorageClient>
41where
42    StorageClient: Storage + Clone + Send + Sync + 'static,
43{
44    state: &'state mut ChainWorkerState<StorageClient>,
45    succeeded: bool,
46}
47
48impl<'state, StorageClient> ChainWorkerStateWithAttemptedChanges<'state, StorageClient>
49where
50    StorageClient: Storage + Clone + Send + Sync + 'static,
51{
52    /// Creates a new [`ChainWorkerStateWithAttemptedChanges`] instance to change the
53    /// `state`.
54    pub(super) async fn new(state: &'state mut ChainWorkerState<StorageClient>) -> Self {
55        assert!(
56            !state.chain.has_pending_changes().await,
57            "`ChainStateView` has unexpected leftover changes"
58        );
59
60        ChainWorkerStateWithAttemptedChanges {
61            state,
62            succeeded: false,
63        }
64    }

66    /// Processes a leader timeout issued for this multi-owner chain.
67    pub(super) async fn process_timeout(
68        &mut self,
69        certificate: TimeoutCertificate,
70    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
71        // Check that the chain is active and ready for this timeout.
72        // Verify the certificate. Returns a catch-all error to make client code more robust.
73        self.state.ensure_is_active().await?;
74        let (chain_epoch, committee) = self.state.chain.current_committee()?;
75        ensure!(
76            certificate.inner().epoch() == chain_epoch,
77            WorkerError::InvalidEpoch {
78                chain_id: certificate.inner().chain_id(),
79                chain_epoch,
80                epoch: certificate.inner().epoch()
81            }
82        );
83        certificate.check(committee)?;
84        let mut actions = NetworkActions::default();
85        if self
86            .state
87            .chain
88            .tip_state
89            .get()
90            .already_validated_block(certificate.inner().height())?
91        {
92            return Ok((
93                ChainInfoResponse::new(&self.state.chain, self.state.config.key_pair()),
94                actions,
95            ));
96        }
97        let old_round = self.state.chain.manager.current_round();
98        let timeout_chain_id = certificate.inner().chain_id();
99        let timeout_height = certificate.inner().height();
100        self.state
101            .chain
102            .manager
103            .handle_timeout_certificate(certificate, self.state.storage.clock().current_time());
104        let round = self.state.chain.manager.current_round();
105        if round > old_round {
106            actions.notifications.push(Notification {
107                chain_id: timeout_chain_id,
108                reason: Reason::NewRound {
109                    height: timeout_height,
110                    round,
111                },
112            })
113        }
114        let info = ChainInfoResponse::new(&self.state.chain, self.state.config.key_pair());
115        self.save().await?;
116        Ok((info, actions))
117    }

119    /// Tries to load all blobs published in this proposal.
120    ///
121    /// If they cannot be found, it creates an entry in `pending_proposed_blobs` so they can be
122    /// submitted one by one.
123    pub(super) async fn load_proposal_blobs(
124        &mut self,
125        proposal: &BlockProposal,
126    ) -> Result<Vec<Blob>, WorkerError> {
127        let BlockProposal {
128            content:
129                ProposalContent {
130                    block,
131                    round,
132                    outcome: _,
133                },
134            public_key,
135            original_proposal,
136            signature: _,
137        } = proposal;
138
139        let owner = AccountOwner::from(*public_key);
140        let mut maybe_blobs = self
141            .state
142            .maybe_get_required_blobs(proposal.required_blob_ids(), None)
143            .await?;
144        let missing_blob_ids = super::missing_blob_ids(&maybe_blobs);
145        if !missing_blob_ids.is_empty() {
146            let chain = &mut self.state.chain;
147            if chain.ownership().open_multi_leader_rounds {
148                // TODO(#3203): Allow multiple pending proposals on permissionless chains.
149                chain.pending_proposed_blobs.clear();
150            }
151            let validated = matches!(original_proposal, Some(OriginalProposal::Regular { .. }));
152            chain
153                .pending_proposed_blobs
154                .try_load_entry_mut(&owner)
155                .await?
156                .update(*round, validated, maybe_blobs)
157                .await?;
158            self.save().await?;
159            return Err(WorkerError::BlobsNotFound(missing_blob_ids));
160        }
161        let published_blobs = block
162            .published_blob_ids()
163            .iter()
164            .filter_map(|blob_id| maybe_blobs.remove(blob_id).flatten())
165            .collect::<Vec<_>>();
166        Ok(published_blobs)
167    }

169    /// Votes for a block proposal for the next block for this chain.
170    pub(super) async fn vote_for_block_proposal(
171        &mut self,
172        proposal: BlockProposal,
173        outcome: BlockExecutionOutcome,
174        local_time: Timestamp,
175    ) -> Result<(), WorkerError> {
176        // Create the vote and store it in the chain state.
177        let block = outcome.with(proposal.content.block.clone());
178        let created_blobs: BTreeMap<_, _> = block.iter_created_blobs().collect();
179        let blobs = self
180            .state
181            .get_required_blobs(proposal.expected_blob_ids(), &created_blobs)
182            .await?;
183        let key_pair = self.state.config.key_pair();
184        let manager = &mut self.state.chain.manager;
185        match manager.create_vote(proposal, block, key_pair, local_time, blobs)? {
186            // Cache the value we voted on, so the client doesn't have to send it again.
187            Some(Either::Left(vote)) => {
188                self.state
189                    .block_values
190                    .insert(Cow::Borrowed(vote.value.inner()));
191            }
192            Some(Either::Right(vote)) => {
193                self.state
194                    .block_values
195                    .insert(Cow::Borrowed(vote.value.inner()));
196            }
197            None => (),
198        }
199        self.save().await?;
200        Ok(())
201    }

203    /// Processes a validated block issued for this multi-owner chain.
204    pub(super) async fn process_validated_block(
205        &mut self,
206        certificate: ValidatedBlockCertificate,
207    ) -> Result<(ChainInfoResponse, NetworkActions, bool), WorkerError> {
208        let block = certificate.block();
209
210        let header = &block.header;
211        let height = header.height;
212        // Check that the chain is active and ready for this validated block.
213        // Verify the certificate. Returns a catch-all error to make client code more robust.
214        self.state.ensure_is_active().await?;
215        let (epoch, committee) = self.state.chain.current_committee()?;
216        check_block_epoch(epoch, header.chain_id, header.epoch)?;
217        certificate.check(committee)?;
218        let mut actions = NetworkActions::default();
219        let already_committed_block = self
220            .state
221            .chain
222            .tip_state
223            .get()
224            .already_validated_block(height)?;
225        let should_skip_validated_block = || {
226            self.state
227                .chain
228                .manager
229                .check_validated_block(&certificate)
230                .map(|outcome| outcome == manager::Outcome::Skip)
231        };
232        if already_committed_block || should_skip_validated_block()? {
233            // If we just processed the same pending block, return the chain info unchanged.
234            return Ok((
235                ChainInfoResponse::new(&self.state.chain, self.state.config.key_pair()),
236                actions,
237                true,
238            ));
239        }
240
241        self.state
242            .block_values
243            .insert(Cow::Borrowed(certificate.inner().inner()));
244        let required_blob_ids = block.required_blob_ids();
245        let maybe_blobs = self
246            .state
247            .maybe_get_required_blobs(required_blob_ids, Some(&block.created_blobs()))
248            .await?;
249        let missing_blob_ids = super::missing_blob_ids(&maybe_blobs);
250        if !missing_blob_ids.is_empty() {
251            self.state
252                .chain
253                .pending_validated_blobs
254                .update(certificate.round, true, maybe_blobs)
255                .await?;
256            self.save().await?;
257            return Err(WorkerError::BlobsNotFound(missing_blob_ids));
258        }
259        let blobs = maybe_blobs
260            .into_iter()
261            .filter_map(|(blob_id, maybe_blob)| Some((blob_id, maybe_blob?)))
262            .collect();
263        let old_round = self.state.chain.manager.current_round();
264        self.state.chain.manager.create_final_vote(
265            certificate,
266            self.state.config.key_pair(),
267            self.state.storage.clock().current_time(),
268            blobs,
269        )?;
270        let info = ChainInfoResponse::new(&self.state.chain, self.state.config.key_pair());
271        self.save().await?;
272        let round = self.state.chain.manager.current_round();
273        if round > old_round {
274            actions.notifications.push(Notification {
275                chain_id: self.state.chain_id(),
276                reason: Reason::NewRound { height, round },
277            })
278        }
279        Ok((info, actions, false))
280    }

282    /// Processes a confirmed block (aka a commit).
283    pub(super) async fn process_confirmed_block(
284        &mut self,
285        certificate: ConfirmedBlockCertificate,
286        notify_when_messages_are_delivered: Option<oneshot::Sender<()>>,
287    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
288        let block = certificate.block();
289        let height = block.header.height;
290        let chain_id = block.header.chain_id;
291
292        // Check that the chain is active and ready for this confirmation.
293        let tip = self.state.chain.tip_state.get().clone();
294        if tip.next_block_height < height {
295            return Err(WorkerError::MissingEarlierBlocks {
296                current_block_height: tip.next_block_height,
297            });
298        }
299        if tip.next_block_height > height {
300            // We already processed this block.
301            let actions = self.state.create_network_actions().await?;
302            self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
303                .await;
304            let info = ChainInfoResponse::new(&self.state.chain, self.state.config.key_pair());
305            return Ok((info, actions));
306        }
307        let local_time = self.state.storage.clock().current_time();
308        // TODO(#2351): This sets the committee and then checks that committee's signatures.
309        self.state.ensure_is_active().await?;
310        // Verify the certificate.
311        let (epoch, committee) = self.state.chain.current_committee()?;
312        check_block_epoch(epoch, chain_id, block.header.epoch)?;
313        certificate.check(committee)?;
314        // This should always be true for valid certificates.
315        ensure!(
316            tip.block_hash == block.header.previous_block_hash,
317            WorkerError::InvalidBlockChaining
318        );
319
320        let required_blob_ids = block.required_blob_ids();
321        let created_blobs: BTreeMap<_, _> = block.iter_created_blobs().collect();
322        let blobs_result = self
323            .state
324            .get_required_blobs(block.required_blob_ids(), &created_blobs)
325            .await
326            .map(|blobs| blobs.into_values().collect::<Vec<_>>());
327
328        if let Ok(blobs) = &blobs_result {
329            self.state
330                .storage
331                .write_blobs_and_certificate(blobs, &certificate)
332                .await?;
333            let events = block
334                .body
335                .events
336                .iter()
337                .flatten()
338                .map(|event| (event.id(chain_id), event.value.clone()));
339            self.state.storage.write_events(events).await?;
340        }
341
342        // Update the blob state with last used certificate hash.
343        let blob_state = certificate.value().to_blob_state();
344        let overwrite = blobs_result.is_ok(); // Overwrite only if we wrote the certificate.
345        let blob_ids = required_blob_ids.into_iter().collect::<Vec<_>>();
346        self.state
347            .storage
348            .maybe_write_blob_states(&blob_ids, blob_state, overwrite)
349            .await?;
350        let mut blobs = blobs_result?
351            .into_iter()
352            .map(|blob| (blob.id(), blob))
353            .collect::<BTreeMap<_, _>>();
354        let published_blobs = block
355            .published_blob_ids()
356            .iter()
357            .filter_map(|blob_id| blobs.remove(blob_id))
358            .collect::<Vec<_>>();
359
360        // Execute the block and update inboxes.
361        let chain = &mut self.state.chain;
362        chain
363            .remove_bundles_from_inboxes(block.header.timestamp, &block.body.incoming_bundles)
364            .await?;
365        let oracle_responses = Some(block.body.oracle_responses.clone());
366        let (proposed_block, outcome) = block.clone().into_proposal();
367        let verified_outcome = if let Some(execution_state) =
368            self.state.execution_state_cache.remove(&outcome.state_hash)
369        {
370            chain.execution_state = execution_state;
371            outcome.clone()
372        } else {
373            chain
374                .execute_block(
375                    &proposed_block,
376                    local_time,
377                    None,
378                    &published_blobs,
379                    oracle_responses,
380                )
381                .await?
382        };
383        // We should always agree on the messages and state hash.
384        ensure!(
385            outcome == verified_outcome,
386            WorkerError::IncorrectOutcome {
387                submitted: Box::new(outcome),
388                computed: Box::new(verified_outcome),
389            }
390        );
391        // Update the rest of the chain state.
392        chain
393            .apply_confirmed_block(certificate.value(), local_time)
394            .await?;
395        self.state
396            .track_newly_created_chains(&proposed_block, &outcome);
397        let mut actions = self.state.create_network_actions().await?;
398        trace!("Processed confirmed block {height} on chain {chain_id:.8}");
399        let hash = certificate.hash();
400        actions.notifications.push(Notification {
401            chain_id,
402            reason: Reason::NewBlock { height, hash },
403        });
404        // Persist chain.
405        self.save().await?;
406
407        self.state
408            .block_values
409            .insert(Cow::Owned(certificate.into_inner().into_inner()));
410
411        self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
412            .await;
413        let info = ChainInfoResponse::new(&self.state.chain, self.state.config.key_pair());
414
415        Ok((info, actions))
416    }

418    /// Schedules a notification for when cross-chain messages are delivered up to the given
419    /// `height`.
420    #[instrument(level = "trace", skip(self, notify_when_messages_are_delivered))]
421    async fn register_delivery_notifier(
422        &mut self,
423        height: BlockHeight,
424        actions: &NetworkActions,
425        notify_when_messages_are_delivered: Option<oneshot::Sender<()>>,
426    ) {
427        if let Some(notifier) = notify_when_messages_are_delivered {
428            if actions
429                .cross_chain_requests
430                .iter()
431                .any(|request| request.has_messages_lower_or_equal_than(height))
432            {
433                self.state.delivery_notifier.register(height, notifier);
434            } else {
435                // No need to wait. Also, cross-chain requests may not trigger the
436                // notifier later, even if we register it.
437                if let Err(()) = notifier.send(()) {
438                    warn!("Failed to notify message delivery to caller");
439                }
440            }
441        }
442    }

444    /// Updates the chain's inboxes, receiving messages from a cross-chain update.
445    pub(super) async fn process_cross_chain_update(
446        &mut self,
447        origin: ChainId,
448        bundles: Vec<(Epoch, MessageBundle)>,
449    ) -> Result<Option<BlockHeight>, WorkerError> {
450        // Only process certificates with relevant heights and epochs.
451        let next_height_to_receive = self
452            .state
453            .chain
454            .next_block_height_to_receive(&origin)
455            .await?;
456        let last_anticipated_block_height = self
457            .state
458            .chain
459            .last_anticipated_block_height(&origin)
460            .await?;
461        let helper = CrossChainUpdateHelper::new(&self.state.config, &self.state.chain);
462        let recipient = self.state.chain_id();
463        let bundles = helper.select_message_bundles(
464            &origin,
465            recipient,
466            next_height_to_receive,
467            last_anticipated_block_height,
468            bundles,
469        )?;
470        let Some(last_updated_height) = bundles.last().map(|bundle| bundle.height) else {
471            return Ok(None);
472        };
473        // Process the received messages in certificates.
474        let local_time = self.state.storage.clock().current_time();
475        let mut previous_height = None;
476        for bundle in bundles {
477            let add_to_received_log = previous_height != Some(bundle.height);
478            previous_height = Some(bundle.height);
479            // Update the staged chain state with the received block.
480            self.state
481                .chain
482                .receive_message_bundle(&origin, bundle, local_time, add_to_received_log)
483                .await?;
484        }
485        if !self.state.config.allow_inactive_chains && !self.state.chain.is_active() {
486            // Refuse to create a chain state if the chain is still inactive by
487            // now. Accordingly, do not send a confirmation, so that the
488            // cross-chain update is retried later.
489            warn!(
490                "Refusing to deliver messages to {recipient:?} from {origin:?} \
491                at height {last_updated_height} because the recipient is still inactive",
492            );
493            return Ok(None);
494        }
495        // Save the chain.
496        self.save().await?;
497        Ok(Some(last_updated_height))
498    }

500    /// Handles the cross-chain request confirming that the recipient was updated.
501    pub(super) async fn confirm_updated_recipient(
502        &mut self,
503        recipient: ChainId,
504        latest_height: BlockHeight,
505    ) -> Result<(), WorkerError> {
506        let fully_delivered = self
507            .state
508            .chain
509            .mark_messages_as_received(&recipient, latest_height)
510            .await?
511            && self
512                .state
513                .all_messages_to_tracked_chains_delivered_up_to(latest_height)
514                .await?;
515
516        self.save().await?;
517
518        if fully_delivered {
519            self.state.delivery_notifier.notify(latest_height);
520        }
521
522        Ok(())
523    }

553    /// Votes for falling back to a public chain.
554    pub(super) async fn vote_for_fallback(&mut self) -> Result<(), WorkerError> {
555        let chain = &mut self.state.chain;
556        if let (epoch, Some(entry)) = (
557            chain.execution_state.system.epoch.get(),
558            chain.unskippable_bundles.front(),
559        ) {
560            let elapsed = self
561                .state
562                .storage
563                .clock()
564                .current_time()
565                .delta_since(entry.seen);
566            if elapsed >= chain.ownership().timeout_config.fallback_duration {
567                let chain_id = chain.chain_id();
568                let height = chain.tip_state.get().next_block_height;
569                let key_pair = self.state.config.key_pair();
570                if chain
571                    .manager
572                    .vote_fallback(chain_id, height, *epoch, key_pair)
573                {
574                    self.save().await?;
575                }
576            }
577        }
578        Ok(())
579    }

581    pub(super) async fn handle_pending_blob(
582        &mut self,
583        blob: Blob,
584    ) -> Result<ChainInfoResponse, WorkerError> {
585        let mut was_expected = self
586            .state
587            .chain
588            .pending_validated_blobs
589            .maybe_insert(&blob)
590            .await?;
591        for (_, mut pending_blobs) in self
592            .state
593            .chain
594            .pending_proposed_blobs
595            .try_load_all_entries_mut()
596            .await?
597        {
598            if !pending_blobs.validated.get() {
599                let (_, committee) = self.state.chain.current_committee()?;
600                let policy = committee.policy();
601                policy
602                    .check_blob_size(blob.content())
603                    .with_execution_context(ChainExecutionContext::Block)?;
604                ensure!(
605                    u64::try_from(pending_blobs.pending_blobs.count().await?)
606                        .is_ok_and(|count| count < policy.maximum_published_blobs),
607                    WorkerError::TooManyPublishedBlobs(policy.maximum_published_blobs)
608                );
609            }
610            was_expected = was_expected || pending_blobs.maybe_insert(&blob).await?;
611        }
612        ensure!(was_expected, WorkerError::UnexpectedBlob);
613        self.save().await?;
614        Ok(ChainInfoResponse::new(
615            &self.state.chain,
616            self.state.config.key_pair(),
617        ))
618    }

620    /// Stores the chain state in persistent storage.
621    ///
622    /// Waits until the [`ChainStateView`] is no longer shared before persisting the changes.
623    async fn save(&mut self) -> Result<(), WorkerError> {
624        // SAFETY: this is the only place a write-lock is acquired, and read-locks are acquired in
625        // the `chain_state_view` method, which has a `&mut self` receiver like this `save` method.
626        // That means that when the write-lock is acquired, no readers will be waiting to acquire
627        // the lock. This is important because otherwise readers could have a stale view of the
628        // chain state.
629        let maybe_shared_chain_view = self.state.shared_chain_view.take();
630        let _maybe_write_guard = match &maybe_shared_chain_view {
631            Some(shared_chain_view) => Some(shared_chain_view.write().await),
632            None => None,
633        };
634
635        self.state.chain.save().await?;
636        self.succeeded = true;
637        Ok(())
638    }

641impl<StorageClient> Drop for ChainWorkerStateWithAttemptedChanges<'_, StorageClient>
642where
643    StorageClient: Storage + Clone + Send + Sync + 'static,
644{
645    fn drop(&mut self) {
646        if !self.succeeded {
647            self.state.chain.rollback();
648        }
649    }

652/// Helper type for handling cross-chain updates.
653pub(crate) struct CrossChainUpdateHelper<'a> {
654    pub allow_messages_from_deprecated_epochs: bool,
655    pub current_epoch: Epoch,
656    pub committees: &'a BTreeMap<Epoch, Committee>,
657}
658
659impl<'a> CrossChainUpdateHelper<'a> {
660    /// Creates a new [`CrossChainUpdateHelper`].
661    pub fn new<C>(config: &ChainWorkerConfig, chain: &'a ChainStateView<C>) -> Self
662    where
663        C: Context + Clone + Send + Sync + 'static,
664    {
665        CrossChainUpdateHelper {
666            allow_messages_from_deprecated_epochs: config.allow_messages_from_deprecated_epochs,
667            current_epoch: *chain.execution_state.system.epoch.get(),
668            committees: chain.execution_state.system.committees.get(),
669        }
670    }
671
672    /// Checks basic invariants and deals with repeated heights and deprecated epochs.
673    /// * Returns a range of message bundles that are both new to us and not relying on
674    ///   an untrusted set of validators.
675    /// * In the case of validators, if the epoch(s) of the highest bundles are not
676    ///   trusted, we only accept bundles that contain messages that were already
677    ///   executed by anticipation (i.e. received in certified blocks).
678    /// * Basic invariants are checked for good measure. We still crucially trust
679    ///   the worker of the sending chain to have verified and executed the blocks
680    ///   correctly.
681    pub fn select_message_bundles(
682        &self,
683        origin: &'a ChainId,
684        recipient: ChainId,
685        next_height_to_receive: BlockHeight,
686        last_anticipated_block_height: Option<BlockHeight>,
687        mut bundles: Vec<(Epoch, MessageBundle)>,
688    ) -> Result<Vec<MessageBundle>, WorkerError> {
689        let mut latest_height = None;
690        let mut skipped_len = 0;
691        let mut trusted_len = 0;
692        for (i, (epoch, bundle)) in bundles.iter().enumerate() {
693            // Make sure that heights are not decreasing.
694            ensure!(
695                latest_height <= Some(bundle.height),
696                WorkerError::InvalidCrossChainRequest
697            );
698            latest_height = Some(bundle.height);
699            // Check if the block has been received already.
700            if bundle.height < next_height_to_receive {
701                skipped_len = i + 1;
702            }
703            // Check if the height is trusted or the epoch is trusted.
704            if self.allow_messages_from_deprecated_epochs
705                || Some(bundle.height) <= last_anticipated_block_height
706                || *epoch >= self.current_epoch
707                || self.committees.contains_key(epoch)
708            {
709                trusted_len = i + 1;
710            }
711        }
712        if skipped_len > 0 {
713            let (_, sample_bundle) = &bundles[skipped_len - 1];
714            debug!(
715                "Ignoring repeated messages to {recipient:.8} from {origin:} at height {}",
716                sample_bundle.height,
717            );
718        }
719        if skipped_len < bundles.len() && trusted_len < bundles.len() {
720            let (sample_epoch, sample_bundle) = &bundles[trusted_len];
721            warn!(
722                "Refusing messages to {recipient:.8} from {origin:} at height {} \
723                 because the epoch {} is not trusted any more",
724                sample_bundle.height, sample_epoch,
725            );
726        }
727        let bundles = if skipped_len < trusted_len {
728            bundles
729                .drain(skipped_len..trusted_len)
730                .map(|(_, bundle)| bundle)
731                .collect()
732        } else {
733            vec![]
734        };
735        Ok(bundles)
736    }
737}
