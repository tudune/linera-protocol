114        Ok(self
115            .shared_chain_view
116            .as_ref()
117            .expect("`shared_chain_view` should be initialized above")
118            .clone()
119            .read_owned()
120            .await)
121    }

197    /// Handles a proposal for the next block for this chain.
198    #[tracing::instrument(level = "debug", skip(self))]
199    pub(super) async fn handle_block_proposal(
200        &mut self,
201        proposal: BlockProposal,
202    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
203        self.ensure_is_active().await?;
204        if ChainWorkerStateWithTemporaryChanges::new(&mut *self)
205            .await
206            .check_proposed_block(&proposal)
207            .await?
208            == manager::Outcome::Skip
209        {
210            // Skipping: We already voted for this block.
211            let info = ChainInfoResponse::new(&self.chain, self.config.key_pair());
212            return Ok((info, NetworkActions::default()));
213        };
214        let published_blobs = ChainWorkerStateWithAttemptedChanges::new(&mut *self)
215            .await
216            .load_proposal_blobs(&proposal)
217            .await?;
218        let validation_outcome = ChainWorkerStateWithTemporaryChanges::new(self)
219            .await
220            .validate_proposal_content(&proposal.content, &published_blobs)
221            .await?;
222
223        let actions = if let Some((outcome, local_time)) = validation_outcome {
224            ChainWorkerStateWithAttemptedChanges::new(&mut *self)
225                .await
226                .vote_for_block_proposal(proposal, outcome, local_time)
227                .await?;
228            // Trigger any outgoing cross-chain messages that haven't been confirmed yet.
229            self.create_network_actions().await?
230        } else {
231            // If we just processed the same pending block, return the chain info unchanged.
232            NetworkActions::default()
233        };
234
235        let info = ChainInfoResponse::new(&self.chain, self.config.key_pair());
236        Ok((info, actions))
237    }

289    /// Handles a [`ChainInfoQuery`], potentially voting on the next block.
290    #[tracing::instrument(level = "debug", skip(self))]
291    pub(super) async fn handle_chain_info_query(
292        &mut self,
293        query: ChainInfoQuery,
294    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
295        if query.request_leader_timeout {
296            ChainWorkerStateWithAttemptedChanges::new(&mut *self)
297                .await
298                .vote_for_leader_timeout()
299                .await?;
300        }
301        if query.request_fallback {
302            ChainWorkerStateWithAttemptedChanges::new(&mut *self)
303                .await
304                .vote_for_fallback()
305                .await?;
306        }
307        let response = ChainWorkerStateWithTemporaryChanges::new(self)
308            .await
309            .prepare_chain_info_response(query)
310            .await?;
311        // Trigger any outgoing cross-chain messages that haven't been confirmed yet.
312        let actions = self.create_network_actions().await?;
313        Ok((response, actions))
314    }

316    /// Returns the requested blob, if it belongs to the current locking block or pending proposal.

318        if let Some(blob) = self.chain.manager.pending_blob(&blob_id).await? {
319            return Ok(blob);

347    /// Reads the blobs from the chain manager or from storage. Returns an error if any are
348    /// missing.
349    async fn get_required_blobs(
350        &self,
351        required_blob_ids: impl IntoIterator<Item = BlobId>,
352        created_blobs: &BTreeMap<BlobId, Blob>,
353    ) -> Result<BTreeMap<BlobId, Blob>, WorkerError> {
354        let maybe_blobs = self
355            .maybe_get_required_blobs(required_blob_ids, Some(created_blobs))
356            .await?;
357        let not_found_blob_ids = missing_blob_ids(&maybe_blobs);
358        ensure!(
359            not_found_blob_ids.is_empty(),
360            WorkerError::BlobsNotFound(not_found_blob_ids)
361        );
362        Ok(maybe_blobs
363            .into_iter()
364            .filter_map(|(blob_id, maybe_blob)| Some((blob_id, maybe_blob?)))
365            .collect())
366    }

368    /// Tries to read the blobs from the chain manager or storage. Returns `None` if not found.
369    async fn maybe_get_required_blobs(
370        &self,
371        blob_ids: impl IntoIterator<Item = BlobId>,
372        created_blobs: Option<&BTreeMap<BlobId, Blob>>,
373    ) -> Result<BTreeMap<BlobId, Option<Blob>>, WorkerError> {
374        let mut maybe_blobs = BTreeMap::from_iter(blob_ids.into_iter().zip(iter::repeat(None)));
375
376        for (blob_id, maybe_blob) in &mut maybe_blobs {
377            if let Some(blob) = created_blobs.and_then(|blob_map| blob_map.get(blob_id)) {
378                *maybe_blob = Some(blob.clone());
379            } else if let Some(blob) = self.chain.manager.pending_blob(blob_id).await? {
380                *maybe_blob = Some(blob);
381            } else if let Some(blob) = self.chain.pending_validated_blobs.get(blob_id).await? {
382                *maybe_blob = Some(blob);
383            } else {
384                for (_, pending_blobs) in self
385                    .chain
386                    .pending_proposed_blobs
387                    .try_load_all_entries()
388                    .await?
389                {
390                    if let Some(blob) = pending_blobs.get(blob_id).await? {
391                        *maybe_blob = Some(blob);
392                        break;
393                    }
394                }
395            }
396        }
397        let missing_blob_ids = missing_blob_ids(&maybe_blobs);
398        let blobs_from_storage = self.storage.read_blobs(&missing_blob_ids).await?;
399        for (blob_id, maybe_blob) in missing_blob_ids.into_iter().zip(blobs_from_storage) {
400            maybe_blobs.insert(blob_id, maybe_blob);
401        }
402        Ok(maybe_blobs)
403    }

405    /// Adds any newly created chains to the set of `tracked_chains`, if the parent chain is
406    /// also tracked.
407    ///
408    /// Chains that are not tracked are usually processed only because they sent some message
409    /// to one of the tracked chains. In most use cases, their children won't be of interest.
410    fn track_newly_created_chains(
411        &self,
412        proposed_block: &ProposedBlock,
413        outcome: &BlockExecutionOutcome,
414    ) {
415        if let Some(tracked_chains) = self.tracked_chains.as_ref() {
416            if !tracked_chains
417                .read()
418                .expect("Panics should not happen while holding a lock to `tracked_chains`")
419                .contains(&proposed_block.chain_id)
420            {
421                return; // The parent chain is not tracked; don't track the child.
422            }
423            let new_chain_ids = outcome
424                .created_blobs_ids()
425                .into_iter()
426                .filter(|blob_id| blob_id.blob_type == BlobType::ChainDescription)
427                .map(|blob_id| ChainId(blob_id.hash));
428
429            tracked_chains
430                .write()
431                .expect("Panics should not happen while holding a lock to `tracked_chains`")
432                .extend(new_chain_ids);
433        }
434    }

436    /// Loads pending cross-chain requests.
437    async fn create_network_actions(&self) -> Result<NetworkActions, WorkerError> {
438        let mut heights_by_recipient = BTreeMap::<_, Vec<_>>::new();
439        let mut targets = self.chain.outboxes.indices().await?;
440        if let Some(tracked_chains) = self.tracked_chains.as_ref() {
441            let tracked_chains = tracked_chains
442                .read()
443                .expect("Panics should not happen while holding a lock to `tracked_chains`");
444            targets.retain(|target| tracked_chains.contains(target));
445        }
446        let outboxes = self.chain.outboxes.try_load_entries(&targets).await?;
447        for (target, outbox) in targets.into_iter().zip(outboxes) {
448            let outbox = outbox.expect("Only existing outboxes should be referenced by `indices`");
449            let heights = outbox.queue.elements().await?;
450            heights_by_recipient.insert(target, heights);
451        }
452        self.create_cross_chain_requests(heights_by_recipient).await
453    }

455    async fn create_cross_chain_requests(
456        &self,
457        heights_by_recipient: BTreeMap<ChainId, Vec<BlockHeight>>,
458    ) -> Result<NetworkActions, WorkerError> {
459        // Load all the certificates we will need, regardless of the medium.
460        let heights = BTreeSet::from_iter(heights_by_recipient.values().flatten().copied());
461        let heights_usize = heights
462            .iter()
463            .copied()
464            .map(usize::try_from)
465            .collect::<Result<Vec<_>, _>>()?;
466        let hashes = self
467            .chain
468            .confirmed_log
469            .multi_get(heights_usize.clone())
470            .await?
471            .into_iter()
472            .zip(heights_usize)
473            .map(|(maybe_hash, height)| {
474                maybe_hash.ok_or_else(|| ViewError::not_found("confirmed log entry", height))
475            })
476            .collect::<Result<Vec<_>, _>>()?;
477        let certificates = self.storage.read_certificates(hashes).await?;
478        let certificates = heights
479            .into_iter()
480            .zip(certificates)
481            .collect::<HashMap<_, _>>();
482        // For each medium, select the relevant messages.
483        let mut actions = NetworkActions::default();
484        for (recipient, heights) in heights_by_recipient {
485            let mut bundles = Vec::new();
486            for height in heights {
487                let cert = certificates
488                    .get(&height)
489                    .ok_or_else(|| ChainError::InternalError("missing certificates".to_string()))?;
490                bundles.extend(cert.message_bundles_for(recipient));
491            }
492            let request = CrossChainRequest::UpdateRecipient {
493                sender: self.chain.chain_id(),
494                recipient,
495                bundles,
496            };
497            actions.cross_chain_requests.push(request);
498        }
499        Ok(actions)
500    }

502    /// Returns true if there are no more outgoing messages in flight up to the given
503    /// block height.
504    pub async fn all_messages_to_tracked_chains_delivered_up_to(
505        &self,
506        height: BlockHeight,
507    ) -> Result<bool, WorkerError> {
508        if self.chain.all_messages_delivered_up_to(height) {
509            return Ok(true);
510        }
511        let Some(tracked_chains) = self.tracked_chains.as_ref() else {
512            return Ok(false);
513        };
514        let mut targets = self.chain.outboxes.indices().await?;
515        {
516            let tracked_chains = tracked_chains.read().unwrap();
517            targets.retain(|target| tracked_chains.contains(target));
518        }
519        let outboxes = self.chain.outboxes.try_load_entries(&targets).await?;
520        for outbox in outboxes {
521            let outbox = outbox.expect("Only existing outboxes should be referenced by `indices`");
522            let front = outbox.queue.front();
523            if front.is_some_and(|key| *key <= height) {
524                return Ok(false);
525            }
526        }
527        Ok(true)
528    }

542/// Returns the keys whose value is `None`.
543fn missing_blob_ids(maybe_blobs: &BTreeMap<BlobId, Option<Blob>>) -> Vec<BlobId> {
544    maybe_blobs
545        .iter()
546        .filter(|(_, maybe_blob)| maybe_blob.is_none())
547        .map(|(blob_id, _)| *blob_id)
548        .collect()
549}

551/// Returns an error if the block is not at the expected epoch.
552fn check_block_epoch(
553    chain_epoch: Epoch,
554    block_chain: ChainId,
555    block_epoch: Epoch,
556) -> Result<(), WorkerError> {
557    ensure!(
558        block_epoch == chain_epoch,
559        WorkerError::InvalidEpoch {
560            chain_id: block_chain,
561            epoch: block_epoch,
562            chain_epoch
563        }
564    );
565    Ok(())
