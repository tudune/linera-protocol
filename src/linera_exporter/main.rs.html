19#[derive(thiserror::Error, Debug)]
20pub(crate) enum ExporterError {
21    #[error("received an invalid notification.")]
22    BadNotification,
23
24    #[error("unable to load the exporter state")]
25    StateError(ViewError),

34/// Options for running the linera block exporter.
35#[derive(clap::Parser, Debug, Clone)]
36#[command(
37    name = "Linera Exporter",
38    version = linera_version::VersionInfo::default_clap_str(),
39)]
40struct ExporterOptions {
41    /// Path to the TOML file describing the configuration for the block exporter.
42    config_path: PathBuf,
43
44    /// Storage configuration for the blockchain history, chain states and binary blobs.
45    #[arg(long = "storage")]
46    storage_config: StorageConfigNamespace,
47
48    /// The number of Tokio worker threads to use.
49    #[arg(long)]
50    tokio_threads: Option<usize>,
51
52    /// The maximal number of simultaneous queries to the database
53    #[arg(long)]
54    max_concurrent_queries: Option<usize>,
55
56    /// The maximal number of stream queries to the database
57    #[arg(long, default_value = "10")]
58    max_stream_queries: usize,
59
60    /// The maximal memory used in the storage cache.
61    #[arg(long, default_value = "10000000")]
62    max_cache_size: usize,
63
64    /// The maximal size of an entry in the storage cache.
65    #[arg(long, default_value = "1000000")]
66    max_entry_size: usize,
67
68    /// The maximal number of entries in the storage cache.
69    #[arg(long, default_value = "1000")]
70    max_cache_entries: usize,
71
72    /// Path to the file describing the initial user chains (aka genesis state)
73    #[arg(long = "genesis")]
74    genesis_config_path: PathBuf,
75}
76
77fn main() -> Result<()> {
78    linera_base::tracing::init("linera-exporter");
79    let options = <ExporterOptions as clap::Parser>::parse();
80    options.run()
81}
82
83impl ExporterOptions {
84    fn run(&self) -> anyhow::Result<()> {
85        let config_string = fs_err::read_to_string(&self.config_path)
86            .expect("Unable to read the configuration file");
87        let config: BlockExporterConfig =
88            toml::from_str(&config_string).expect("Invalid configuration file format");
89
90        let storage_cache_config = StorageCacheConfig {
91            max_cache_size: self.max_cache_size,
92            max_entry_size: self.max_entry_size,
93            max_cache_entries: self.max_cache_entries,
94        };
95        let common_config = CommonStoreConfig {
96            max_concurrent_queries: self.max_concurrent_queries,
97            max_stream_queries: self.max_stream_queries,
98            storage_cache_config,
99            replication_factor: 1,
100        };
101
102        let mut runtime_builder = match self.tokio_threads {
103            None | Some(1) => tokio::runtime::Builder::new_current_thread(),
104            Some(worker_threads) => {
105                let mut builder = tokio::runtime::Builder::new_multi_thread();
106                builder.worker_threads(worker_threads);
107                builder
108            }
109        };
110
111        let genesis_config: GenesisConfig = util::read_json(&self.genesis_config_path)?;
112
113        let future = async {
114            let context =
115                ExporterContext::new(config.id, config.service_config, config.destination_config);
116            let storage_config = self
117                .storage_config
118                .add_common_config(common_config)
119                .await
120                .unwrap();
121            storage_config
122                .run_with_storage(&genesis_config, None, context)
123                .boxed()
124                .await
125        };
126
127        let runtime = runtime_builder.enable_all().build()?;
128        runtime.block_on(future)?.map_err(|e| e.into())
129    }
130}
