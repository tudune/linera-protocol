226trait BatchExt {
227    fn add_blob(&mut self, blob: &Blob) -> Result<(), ViewError>;
228
229    fn add_blob_state(&mut self, blob_id: BlobId, blob_state: &BlobState) -> Result<(), ViewError>;
230
231    fn add_certificate(&mut self, certificate: &ConfirmedBlockCertificate)
232        -> Result<(), ViewError>;
233
234    fn add_event(&mut self, event_id: EventId, value: Vec<u8>) -> Result<(), ViewError>;
235
236    fn add_network_description(
237        &mut self,
238        information: &NetworkDescription,
239    ) -> Result<(), ViewError>;
240}
241
242impl BatchExt for Batch {
243    fn add_blob(&mut self, blob: &Blob) -> Result<(), ViewError> {
244        #[cfg(with_metrics)]
245        metrics::WRITE_BLOB_COUNTER.with_label_values(&[]).inc();
246        let blob_key = bcs::to_bytes(&BaseKey::Blob(blob.id()))?;
247        self.put_key_value_bytes(blob_key.to_vec(), blob.bytes().to_vec());
248        Ok(())
249    }
250
251    fn add_blob_state(&mut self, blob_id: BlobId, blob_state: &BlobState) -> Result<(), ViewError> {
252        let blob_state_key = bcs::to_bytes(&BaseKey::BlobState(blob_id))?;
253        self.put_key_value(blob_state_key.to_vec(), blob_state)?;
254        Ok(())
255    }
256
257    fn add_certificate(
258        &mut self,
259        certificate: &ConfirmedBlockCertificate,
260    ) -> Result<(), ViewError> {
261        #[cfg(with_metrics)]
262        metrics::WRITE_CERTIFICATE_COUNTER
263            .with_label_values(&[])
264            .inc();
265        let hash = certificate.hash();
266        let cert_key = bcs::to_bytes(&BaseKey::Certificate(hash))?;
267        let block_key = bcs::to_bytes(&BaseKey::ConfirmedBlock(hash))?;
268        self.put_key_value(cert_key.to_vec(), &certificate.lite_certificate())?;
269        self.put_key_value(block_key.to_vec(), certificate.value())?;
270        Ok(())
271    }
272
273    fn add_event(&mut self, event_id: EventId, value: Vec<u8>) -> Result<(), ViewError> {
274        #[cfg(with_metrics)]
275        metrics::WRITE_EVENT_COUNTER.with_label_values(&[]).inc();
276        let event_key = bcs::to_bytes(&BaseKey::Event(event_id))?;
277        self.put_key_value_bytes(event_key.to_vec(), value);
278        Ok(())
279    }
280
281    fn add_network_description(
282        &mut self,
283        information: &NetworkDescription,
284    ) -> Result<(), ViewError> {
285        #[cfg(with_metrics)]
286        metrics::WRITE_NETWORK_DESCRIPTION
287            .with_label_values(&[])
288            .inc();
289        let key = bcs::to_bytes(&BaseKey::NetworkDescription)?;
290        self.put_key_value(key, information)?;
291        Ok(())
292    }

397/// An implementation of [`DualStoreRootKeyAssignment`] that stores the
398/// chain states into the first store.
399#[derive(Clone, Copy)]
400pub struct ChainStatesFirstAssignment;
401
402impl DualStoreRootKeyAssignment for ChainStatesFirstAssignment {
403    fn assigned_store(root_key: &[u8]) -> Result<StoreInUse, bcs::Error> {
404        if root_key.is_empty() {
405            return Ok(StoreInUse::Second);
406        }
407        let store = match bcs::from_bytes(root_key)? {
408            BaseKey::ChainState(_) => StoreInUse::First,
409            _ => StoreInUse::Second,
410        };
411        Ok(store)
412    }
413}

415/// A `Clock` implementation using the system clock.
416#[derive(Clone)]
417pub struct WallClock;
418
419#[cfg_attr(not(web), async_trait)]
420#[cfg_attr(web, async_trait(?Send))]
421impl Clock for WallClock {
422    fn current_time(&self) -> Timestamp {
423        Timestamp::now()
424    }
425
426    async fn sleep(&self, delta: TimeDelta) {
427        linera_base::time::timer::sleep(delta.as_duration()).await
428    }
429
430    async fn sleep_until(&self, timestamp: Timestamp) {
431        let delta = timestamp.delta_since(Timestamp::now());
432        if delta > TimeDelta::ZERO {
433            self.sleep(delta).await
434        }
435    }
436}

438#[cfg(with_testing)]
439#[derive(Default)]
440struct TestClockInner {
441    time: Timestamp,
442    sleeps: BTreeMap<Reverse<Timestamp>, Vec<oneshot::Sender<()>>>,
443}
444
445#[cfg(with_testing)]
446impl TestClockInner {
447    fn set(&mut self, time: Timestamp) {
448        self.time = time;
449        let senders = self.sleeps.split_off(&Reverse(time));
450        for sender in senders.into_values().flatten() {
451            let _ = sender.send(());
452        }
453    }
454
455    fn add_sleep(&mut self, delta: TimeDelta) -> Receiver<()> {
456        self.add_sleep_until(self.time.saturating_add(delta))
457    }
458
459    fn add_sleep_until(&mut self, time: Timestamp) -> Receiver<()> {
460        let (sender, receiver) = oneshot::channel();
461        if self.time >= time {
462            let _ = sender.send(());
463        } else {
464            self.sleeps.entry(Reverse(time)).or_default().push(sender);
465        }
466        receiver
467    }
468}

470/// A clock implementation that uses a stored number of microseconds and that can be updated
471/// explicitly. All clones share the same time, and setting it in one clone updates all the others.
472#[cfg(with_testing)]
473#[derive(Clone, Default)]
474pub struct TestClock(Arc<std::sync::Mutex<TestClockInner>>);
475
476#[cfg(with_testing)]
477#[cfg_attr(not(web), async_trait)]
478#[cfg_attr(web, async_trait(?Send))]
479impl Clock for TestClock {
480    fn current_time(&self) -> Timestamp {
481        self.lock().time
482    }
483
484    async fn sleep(&self, delta: TimeDelta) {
485        if delta == TimeDelta::ZERO {
486            return;
487        }
488        let receiver = self.lock().add_sleep(delta);
489        let _ = receiver.await;
490    }
491
492    async fn sleep_until(&self, timestamp: Timestamp) {
493        let receiver = self.lock().add_sleep_until(timestamp);
494        let _ = receiver.await;
495    }
496}
497
498#[cfg(with_testing)]
499impl TestClock {
500    /// Creates a new clock with its time set to 0, i.e. the Unix epoch.
501    pub fn new() -> Self {
502        TestClock(Arc::default())
503    }
504
505    /// Sets the current time.
506    pub fn set(&self, time: Timestamp) {
507        self.lock().set(time);
508    }
509
510    /// Advances the current time by the specified delta.
511    pub fn add(&self, delta: TimeDelta) {
512        let mut guard = self.lock();
513        let time = guard.time.saturating_add(delta);
514        guard.set(time);
515    }
516
517    /// Returns the current time according to the test clock.
518    pub fn current_time(&self) -> Timestamp {
519        self.lock().time
520    }
521
522    fn lock(&self) -> std::sync::MutexGuard<TestClockInner> {
523        self.0.lock().expect("poisoned TestClock mutex")
524    }

543    async fn load_chain(
544        &self,
545        chain_id: ChainId,
546    ) -> Result<ChainStateView<Self::Context>, ViewError> {
547        #[cfg(with_metrics)]
548        let _metric = metrics::LOAD_CHAIN_LATENCY.measure_latency();
549        let runtime_context = ChainRuntimeContext {
550            storage: self.clone(),
551            chain_id,
552            execution_runtime_config: self.execution_runtime_config,
553            user_contracts: self.user_contracts.clone(),
554            user_services: self.user_services.clone(),
555        };
556        let root_key = bcs::to_bytes(&BaseKey::ChainState(chain_id))?;
557        let store = self.store.clone_with_root_key(&root_key)?;
558        let context = ViewContext::create_root_context(store, runtime_context).await?;
559        ChainStateView::load(context).await
560    }

562    async fn contains_blob(&self, blob_id: BlobId) -> Result<bool, ViewError> {
563        let blob_key = bcs::to_bytes(&BaseKey::Blob(blob_id))?;
564        let test = self.store.contains_key(&blob_key).await?;
565        #[cfg(with_metrics)]
566        metrics::CONTAINS_BLOB_COUNTER.with_label_values(&[]).inc();
567        Ok(test)
568    }

570    async fn missing_blobs(&self, blob_ids: &[BlobId]) -> Result<Vec<BlobId>, ViewError> {
571        let mut keys = Vec::new();
572        for blob_id in blob_ids {
573            let key = bcs::to_bytes(&BaseKey::Blob(*blob_id))?;
574            keys.push(key);
575        }
576        let results = self.store.contains_keys(keys).await?;
577        let mut missing_blobs = Vec::new();
578        for (blob_id, result) in blob_ids.iter().zip(results) {
579            if !result {
580                missing_blobs.push(*blob_id);
581            }
582        }
583        #[cfg(with_metrics)]
584        metrics::CONTAINS_BLOBS_COUNTER.with_label_values(&[]).inc();
585        Ok(missing_blobs)
586    }

588    async fn contains_blob_state(&self, blob_id: BlobId) -> Result<bool, ViewError> {
589        let blob_key = bcs::to_bytes(&BaseKey::BlobState(blob_id))?;
590        let test = self.store.contains_key(&blob_key).await?;
591        #[cfg(with_metrics)]
592        metrics::CONTAINS_BLOB_STATE_COUNTER
593            .with_label_values(&[])
594            .inc();
595        Ok(test)
596    }
597
598    async fn read_confirmed_block(&self, hash: CryptoHash) -> Result<ConfirmedBlock, ViewError> {
599        let block_key = bcs::to_bytes(&BaseKey::ConfirmedBlock(hash))?;
600        let maybe_value = self.store.read_value::<ConfirmedBlock>(&block_key).await?;
601        #[cfg(with_metrics)]
602        metrics::READ_CONFIRMED_BLOCK_COUNTER
603            .with_label_values(&[])
604            .inc();
605        let value = maybe_value.ok_or_else(|| ViewError::not_found("value for hash", hash))?;
606        Ok(value)
607    }
608
609    async fn read_blob(&self, blob_id: BlobId) -> Result<Option<Blob>, ViewError> {
610        let blob_key = bcs::to_bytes(&BaseKey::Blob(blob_id))?;
611        let maybe_blob_bytes = self.store.read_value_bytes(&blob_key).await?;
612        #[cfg(with_metrics)]
613        metrics::READ_BLOB_COUNTER.with_label_values(&[]).inc();
614        Ok(maybe_blob_bytes.map(|blob_bytes| Blob::new_with_id_unchecked(blob_id, blob_bytes)))
615    }
616
617    async fn read_blobs(&self, blob_ids: &[BlobId]) -> Result<Vec<Option<Blob>>, ViewError> {
618        if blob_ids.is_empty() {
619            return Ok(Vec::new());
620        }
621        let blob_keys = blob_ids
622            .iter()
623            .map(|blob_id| bcs::to_bytes(&BaseKey::Blob(*blob_id)))
624            .collect::<Result<Vec<_>, _>>()?;
625        let maybe_blob_bytes = self.store.read_multi_values_bytes(blob_keys).await?;
626        #[cfg(with_metrics)]
627        metrics::READ_BLOB_COUNTER
628            .with_label_values(&[])
629            .inc_by(blob_ids.len() as u64);
630
631        Ok(blob_ids
632            .iter()
633            .zip(maybe_blob_bytes)
634            .map(|(blob_id, maybe_blob_bytes)| {
635                maybe_blob_bytes.map(|blob_bytes| Blob::new_with_id_unchecked(*blob_id, blob_bytes))
636            })
637            .collect())
638    }

697    async fn maybe_write_blob_state(
698        &self,
699        blob_id: BlobId,
700        blob_state: BlobState,
701    ) -> Result<Epoch, ViewError> {
702        let current_blob_state = self.read_blob_state(blob_id).await?;
703        let (should_write, latest_epoch) = match current_blob_state {
704            Some(current_blob_state) => (
705                current_blob_state.epoch < blob_state.epoch,
706                current_blob_state.epoch.max(blob_state.epoch),
707            ),
708            None => (true, blob_state.epoch),
709        };
710
711        if should_write {
712            self.write_blob_state(blob_id, &blob_state).await?;
713        }
714
715        Ok(latest_epoch)
716    }

718    async fn maybe_write_blob_states(
719        &self,
720        blob_ids: &[BlobId],
721        blob_state: BlobState,
722        overwrite: bool,
723    ) -> Result<Vec<Epoch>, ViewError> {
724        if blob_ids.is_empty() {
725            return Ok(Vec::new());
726        }
727        let blob_state_keys = blob_ids
728            .iter()
729            .map(|blob_id| bcs::to_bytes(&BaseKey::BlobState(*blob_id)))
730            .collect::<Result<_, _>>()?;
731        let maybe_blob_states = self
732            .store
733            .read_multi_values::<BlobState>(blob_state_keys)
734            .await?;
735        let mut latest_epochs = Vec::new();
736        let mut batch = Batch::new();
737        let mut need_write = false;
738        for (maybe_blob_state, blob_id) in maybe_blob_states.iter().zip(blob_ids) {
739            let (should_write, latest_epoch) = match maybe_blob_state {
740                None => (true, blob_state.epoch),
741                Some(current_blob_state) => (
742                    overwrite && current_blob_state.epoch < blob_state.epoch,
743                    current_blob_state.epoch.max(blob_state.epoch),
744                ),
745            };
746            if should_write {
747                batch.add_blob_state(*blob_id, &blob_state)?;
748                need_write = true;
749            }
750            latest_epochs.push(latest_epoch);
751        }
752        if need_write {
753            self.write_batch(batch).await?;
754        }
755        Ok(latest_epochs)
756    }

839        &self,
840        hashes: I,

842        let hashes = hashes.into_iter().collect::<Vec<_>>();
843        if hashes.is_empty() {

845        }

879    async fn read_events_from_index(
880        &self,
881        chain_id: &ChainId,
882        stream_id: &StreamId,
883        start_index: u32,
884    ) -> Result<Vec<IndexAndEvent>, ViewError> {
885        let mut prefix = vec![INDEX_EVENT_ID];
886        prefix.extend(bcs::to_bytes(chain_id).unwrap());
887        prefix.extend(bcs::to_bytes(stream_id).unwrap());
888        let mut keys = Vec::new();
889        let mut indices = Vec::new();
890        for short_key in self.store.find_keys_by_prefix(&prefix).await?.iterator() {
891            let short_key = short_key?;
892            let index = bcs::from_bytes::<u32>(short_key)?;
893            if index >= start_index {
894                let mut key = prefix.clone();
895                key.extend(short_key);
896                keys.push(key);
897                indices.push(index);
898            }
899        }
900        let values = self.store.read_multi_values_bytes(keys).await?;
901        let mut returned_values = Vec::new();
902        for (index, value) in indices.into_iter().zip(values) {
903            let event = value.unwrap();
904            returned_values.push(IndexAndEvent { index, event });
905        }
906        Ok(returned_values)
907    }

954impl<Store, C> DbStorage<Store, C>
955where
956    Store: KeyValueStore + Clone + Send + Sync + 'static,
957    C: Clock,
958    Store::Error: Send + Sync,
959{
960    fn get_keys_for_certificates(hashes: &[CryptoHash]) -> Result<Vec<Vec<u8>>, ViewError> {
961        Ok(hashes
962            .iter()
963            .flat_map(|hash| {
964                let cert_key = bcs::to_bytes(&BaseKey::Certificate(*hash));
965                let block_key = bcs::to_bytes(&BaseKey::ConfirmedBlock(*hash));
966                vec![cert_key, block_key]
967            })
968            .collect::<Result<_, _>>()?)
969    }
970
971    fn deserialize_certificate(
972        pair: &[Option<Vec<u8>>],
973        hash: CryptoHash,
974    ) -> Result<ConfirmedBlockCertificate, ViewError> {
975        let cert_bytes = pair[0]
976            .as_ref()
977            .ok_or_else(|| ViewError::not_found("certificate bytes for hash", hash))?;
978        let value_bytes = pair[1]
979            .as_ref()
980            .ok_or_else(|| ViewError::not_found("value bytes for hash", hash))?;
981        let cert = bcs::from_bytes::<LiteCertificate>(cert_bytes)?;
982        let value = bcs::from_bytes::<ConfirmedBlock>(value_bytes)?;
983        assert_eq!(value.hash(), hash);
984        let certificate = cert
985            .with_value(value)
986            .ok_or(ViewError::InconsistentEntries)?;
987        Ok(certificate)
988    }

990    async fn write_batch(&self, batch: Batch) -> Result<(), ViewError> {
991        self.store.write_batch(batch).await?;
992        Ok(())
993    }
