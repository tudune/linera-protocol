26#[derive(Debug)]
27struct NoMatchingAuthSchemeError(ExploredList);

29impl fmt::Display for NoMatchingAuthSchemeError {
30    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
31        let explored = &self.0;
32
33        // Use the information we have about the auth options that were explored to construct
34        // as helpful of an error message as possible.
35        if explored.items().count() == 0 {
36            return f.write_str(
37                "no auth options are available. This can happen if there's \
38                    a problem with the service model, or if there is a codegen bug.",
39            );
40        }
41        if explored
42            .items()
43            .all(|explored| matches!(explored.result, ExploreResult::NoAuthScheme))
44        {
45            return f.write_str(
46                "no auth schemes are registered. This can happen if there's \
47                    a problem with the service model, or if there is a codegen bug.",
48            );
49        }
50
51        let mut try_add_identity = false;
52        let mut likely_bug = false;
53        f.write_str("failed to select an auth scheme to sign the request with.")?;
54        for item in explored.items() {
55            write!(
56                f,
57                " \"{}\" wasn't a valid option because ",
58                item.scheme_id.inner()
59            )?;
60            f.write_str(match item.result {
61                ExploreResult::NoAuthScheme => {
62                    likely_bug = true;
63                    "no auth scheme was registered for it."
64                }
65                ExploreResult::NoIdentityResolver => {
66                    try_add_identity = true;
67                    "there was no identity resolver for it."
68                }
69                ExploreResult::MissingEndpointConfig => {
70                    likely_bug = true;
71                    "there is auth config in the endpoint config, but this scheme wasn't listed in it \
72                    (see https://github.com/smithy-lang/smithy-rs/discussions/3281 for more details)."
73                }
74                ExploreResult::NotExplored => {
75                    debug_assert!(false, "this should be unreachable");
76                    "<unknown>"
77                }
78            })?;
79        }
80        if try_add_identity {
81            f.write_str(" Be sure to set an identity, such as credentials, auth token, or other identity type that is required for this service.")?;
82        } else if likely_bug {
83            f.write_str(" This is likely a bug.")?;
84        }
85        if explored.truncated {
86            f.write_str(" Note: there were other auth schemes that were evaluated that weren't listed here.")?;
87        }
88
89        Ok(())
90    }
91}

95#[derive(Debug)]
96enum AuthOrchestrationError {
97    MissingEndpointConfig,
98    BadAuthSchemeEndpointConfig(Cow<'static, str>),
99    FailedToResolveEndpoint(BoxError),
100}
101
102impl fmt::Display for AuthOrchestrationError {
103    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
104        match self {
105            // This error is never bubbled up
106            Self::MissingEndpointConfig => f.write_str("missing endpoint config"),
107            Self::BadAuthSchemeEndpointConfig(message) => f.write_str(message),
108            Self::FailedToResolveEndpoint(source) => {
109                write!(f, "failed to resolve endpoint: {}", source)
110            }
111        }
112    }
113}

117pub(super) async fn resolve_identity(
118    runtime_components: &RuntimeComponents,
119    cfg: &ConfigBag,
120) -> Result<(AuthSchemeId, Identity, Option<Endpoint>), BoxError> {
121    let params = cfg
122        .load::<AuthSchemeOptionResolverParams>()
123        .expect("auth scheme option resolver params must be set");
124    let option_resolver = runtime_components.auth_scheme_option_resolver();
125    let options = option_resolver
126        .resolve_auth_scheme_options_v2(params, cfg, runtime_components)
127        .await?;
128
129    trace!(
130        auth_scheme_option_resolver_params = ?params,
131        auth_scheme_options = ?options,
132        "orchestrating auth",
133    );
134
135    let mut explored = ExploredList::default();
136
137    // Iterate over IDs of possibly-supported auth schemes
138    for auth_scheme_option in options {
139        let scheme_id = auth_scheme_option.scheme_id();
140        // For each ID, try to resolve the corresponding auth scheme.
141        if let Some(auth_scheme) = runtime_components.auth_scheme(scheme_id) {
142            // Use the resolved auth scheme to resolve an identity
143            if let Some(identity_resolver) = auth_scheme.identity_resolver(runtime_components) {
144                match legacy_try_resolve_endpoint(runtime_components, cfg, scheme_id).await {
145                    Ok(endpoint) => {
146                        trace!(scheme_id= ?scheme_id, "resolving identity");
147                        let identity_cache = if identity_resolver.cache_location()
148                            == IdentityCacheLocation::RuntimeComponents
149                        {
150                            runtime_components.identity_cache()
151                        } else {
152                            IdentityCache::no_cache()
153                        };
154                        let identity = identity_cache
155                            .resolve_cached_identity(identity_resolver, runtime_components, cfg)
156                            .await?;
157                        trace!(identity = ?identity, "resolved identity");
158                        return Ok((scheme_id.clone(), identity, endpoint));
159                    }
160                    Err(AuthOrchestrationError::MissingEndpointConfig) => {
161                        explored.push(scheme_id.clone(), ExploreResult::MissingEndpointConfig);
162                        continue;
163                    }
164                    Err(AuthOrchestrationError::FailedToResolveEndpoint(source)) => {
165                        // Some negative endpoint tests expect an endpoint resolution error,
166                        // so we need to return it to satisfy them.
167                        return Err(source);
168                    }
169                    Err(other_err) => {
170                        return Err(other_err.into());
171                    }
172                }
173            } else {
174                explored.push(scheme_id.clone(), ExploreResult::NoIdentityResolver);
175            }
176        } else {
177            explored.push(scheme_id.clone(), ExploreResult::NoAuthScheme);
178        }
179    }
180
181    Err(NoMatchingAuthSchemeError(explored).into())
182}

184pub(super) fn sign_request(
185    scheme_id: &AuthSchemeId,
186    identity: &Identity,
187    ctx: &mut InterceptorContext,
188    runtime_components: &RuntimeComponents,
189    cfg: &ConfigBag,
190) -> Result<(), BoxError> {
191    trace!("signing request");
192    let request = ctx.request_mut().expect("set during serialization");
193    let endpoint = cfg
194        .load::<Endpoint>()
195        .expect("endpoint added to config bag by endpoint orchestrator");
196    let auth_scheme = runtime_components
197        .auth_scheme(scheme_id)
198        .ok_or("should be configured")?;
199    let signer = auth_scheme.signer();
200    let auth_scheme_endpoint_config = extract_endpoint_auth_scheme_config(endpoint, scheme_id)?;
201    trace!(
202        signer = ?signer,
203        "signing implementation"
204    );
205    signer.sign_http_request(
206        request,
207        identity,
208        auth_scheme_endpoint_config,
209        runtime_components,
210        cfg,
211    )?;
212    Ok(())
213}

215// Marker indicating the correct resolution order: auth scheme resolution, identity resolution,
216// and then endpoint resolution, as specified in the SRA.
217//
218// This marker is included in the config bag to signify the intended resolution order
219// by design. When the crate was released for GA, the resolution order was reversed
220// (endpoint resolution first, followed by identity resolution). However, we later
221// discovered the order needed correcting without breaking existing SDKs.
222// This marker signals the runtime to support both resolution orders without introducing
223// `aws-smithy-runtime` version 2.x.
224//
225// When this marker is present in the config bag (the default behavior for forward compatibility),
226// `resolve_identity` skips endpoint resolution, ensuring that `try_attempt` follows
227// the correct resolution order: auth scheme → identity → endpoint.
228// If the marker is absent, `try_attempt` continues using
229// the legacy, incorrect resolution order: endpoint → auth scheme → identity.
230#[doc(hidden)]
231#[derive(Clone, Debug)]
232pub struct AuthSchemeAndEndpointOrchestrationV2;
233
234impl Storable for AuthSchemeAndEndpointOrchestrationV2 {
235    type Storer = StoreReplace<Self>;
236}
237
238// Conditionally return an `endpoint` resolved by `SharedEndpointResolver` in `runtime_components`
239// whose `authSchemes` property matches the given `scheme_id`
240//
241// Return the `Ok` variant in the following cases:
242// - If `AuthSchemeAndEndpointOrchestrationV2` is present in the config bag, indicating that the runtime doesn't require the functionality of this function.
243//   In this case, the function short-circuits and returns `Ok(None)`, as no endpoint needs resolution. Instead, a custom `AuthSchemeOptionResolver`
244//   should have resolved it internally for auth scheme selection.
245// - If the runtime uses the legacy auth scheme and endpoint orchestration and resolves an endpoint that matches the given condition,
246//   the function returns `Ok(endpoint)`, which will then be applied to the request.
247async fn legacy_try_resolve_endpoint(
248    runtime_components: &RuntimeComponents,
249    cfg: &ConfigBag,
250    scheme_id: &AuthSchemeId,
251) -> Result<Option<Endpoint>, AuthOrchestrationError> {
252    if cfg.load::<AuthSchemeAndEndpointOrchestrationV2>().is_some() {
253        // The orchestrator uses the correct auth scheme and endpoint resolution order,
254        // and no endpoint needs to be resolved within this function.
255        return Ok(None);
256    }
257
258    let params = cfg
259        .load::<EndpointResolverParams>()
260        .expect("endpoint resolver params must be set");
261
262    tracing::debug!(scheme_id = ?scheme_id, endpoint_params = ?params, "using legacy auth and endpoint orchestration, resolving endpoint for auth scheme selection");
263
264    let endpoint = runtime_components
265        .endpoint_resolver()
266        .resolve_endpoint(params)
267        .await
268        .map_err(AuthOrchestrationError::FailedToResolveEndpoint)?;
269
270    // This line repurposes `extract_endpoint_auth_scheme_config` to check whether
271    // the function returns `Ok` for the given `endpoint`.
272    // Essentially, we verify if the `authSchemes` property of `endpoint` contains `scheme_id`,
273    // which is done by `schemes.iter().find(...).ok_or(...)` within the function.
274    // However, this execution path is only exercised for legacy auth scheme and endpoint orchestration,
275    // so we don't bother refactoring the predicate out of the function.
276    let _ = extract_endpoint_auth_scheme_config(&endpoint, scheme_id)?;
277
278    Ok(Some(endpoint))
279}

281fn extract_endpoint_auth_scheme_config<'a>(
282    endpoint: &'a Endpoint,
283    scheme_id: &AuthSchemeId,
284) -> Result<AuthSchemeEndpointConfig<'a>, AuthOrchestrationError> {
285    // TODO(P96049742): Endpoint config doesn't currently have a concept of optional auth or "no auth", so
286    // we are short-circuiting lookup of endpoint auth scheme config if that is the selected scheme.
287    if scheme_id == &NO_AUTH_SCHEME_ID {
288        return Ok(AuthSchemeEndpointConfig::empty());
289    }
290    let auth_schemes = match endpoint.properties().get("authSchemes") {
291        Some(Document::Array(schemes)) => schemes,
292        // no auth schemes:
293        None => return Ok(AuthSchemeEndpointConfig::empty()),
294        _other => {
295            return Err(AuthOrchestrationError::BadAuthSchemeEndpointConfig(
296                "expected an array for `authSchemes` in endpoint config".into(),
297            ))
298        }
299    };
300    let auth_scheme_config = auth_schemes
301        .iter()
302        .find(|doc| {
303            let config_scheme_id = doc
304                .as_object()
305                .and_then(|object| object.get("name"))
306                .and_then(Document::as_string);
307            config_scheme_id == Some(scheme_id.inner())
308        })
309        .ok_or(AuthOrchestrationError::MissingEndpointConfig)?;
310    Ok(AuthSchemeEndpointConfig::from(Some(auth_scheme_config)))
311}

313#[derive(Debug)]
314enum ExploreResult {
315    NotExplored,
316    NoAuthScheme,
317    NoIdentityResolver,
318    MissingEndpointConfig,
319}

321/// Information about an evaluated auth option.
322/// This should be kept small so it can fit in an array on the stack.
323#[derive(Debug)]
324struct ExploredAuthOption {
325    scheme_id: AuthSchemeId,
326    result: ExploreResult,
327}
328impl Default for ExploredAuthOption {
329    fn default() -> Self {
330        Self {
331            scheme_id: AuthSchemeId::new(""),
332            result: ExploreResult::NotExplored,
333        }
334    }
335}
336
337const MAX_EXPLORED_LIST_LEN: usize = 8;
338
339/// Stack allocated list of explored auth options for error messaging
340#[derive(Default)]
341struct ExploredList {
342    items: [ExploredAuthOption; MAX_EXPLORED_LIST_LEN],
343    len: usize,
344    truncated: bool,
345}
346impl ExploredList {
347    fn items(&self) -> impl Iterator<Item = &ExploredAuthOption> {
348        self.items.iter().take(self.len)
349    }
350
351    fn push(&mut self, scheme_id: AuthSchemeId, result: ExploreResult) {
352        if self.len + 1 >= self.items.len() {
353            self.truncated = true;
354        } else {
355            self.items[self.len] = ExploredAuthOption { scheme_id, result };
356            self.len += 1;
357        }
358    }
359}
360impl fmt::Debug for ExploredList {
361    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
362        f.debug_struct("ExploredList")
363            .field("items", &&self.items[0..self.len])
364            .field("truncated", &self.truncated)
365            .finish()
366    }
367}

708    async fn use_identity_cache() {
709        use crate::client::auth::http::{ApiKeyAuthScheme, ApiKeyLocation};
710        use aws_smithy_runtime_api::client::auth::http::HTTP_API_KEY_AUTH_SCHEME_ID;
711        use aws_smithy_runtime_api::client::identity::http::Token;
712        use aws_smithy_types::body::SdkBody;
713
714        #[derive(Debug)]
715        struct Cache;
716        impl ResolveCachedIdentity for Cache {
717            fn resolve_cached_identity<'a>(
718                &'a self,
719                _resolver: SharedIdentityResolver,
720                _: &'a RuntimeComponents,
721                _config_bag: &'a ConfigBag,
722            ) -> IdentityFuture<'a> {
723                IdentityFuture::ready(Ok(Identity::new(Token::new("cached (pass)", None), None)))
724            }
725        }
726
727        async fn run_test(add_more_to_layer: impl Fn(Layer) -> Layer) {
728            let mut ctx = InterceptorContext::new(Input::doesnt_matter());
729            ctx.enter_serialization_phase();
730            ctx.set_request(
731                http_02x::Request::builder()
732                    .body(SdkBody::empty())
733                    .unwrap()
734                    .try_into()
735                    .unwrap(),
736            );
737            let _ = ctx.take_input();
738            ctx.enter_before_transmit_phase();
739
740            let runtime_components = RuntimeComponentsBuilder::for_tests()
741                .with_auth_scheme(SharedAuthScheme::new(ApiKeyAuthScheme::new(
742                    "result:",
743                    ApiKeyLocation::Header,
744                    "Authorization",
745                )))
746                .with_auth_scheme_option_resolver(Some(SharedAuthSchemeOptionResolver::new(
747                    StaticAuthSchemeOptionResolver::new(vec![HTTP_API_KEY_AUTH_SCHEME_ID]),
748                )))
749                .with_identity_cache(Some(Cache))
750                .with_identity_resolver(
751                    HTTP_API_KEY_AUTH_SCHEME_ID,
752                    SharedIdentityResolver::new(Token::new("uncached (fail)", None)),
753                )
754                .with_endpoint_resolver(Some(SharedEndpointResolver::new(
755                    StaticUriEndpointResolver::http_localhost(8080),

757                .build()
758                .unwrap();
759            let mut layer = Layer::new("test");

803    #[test]
804    fn friendly_error_messages() {
805        let err = NoMatchingAuthSchemeError(ExploredList::default());
806        assert_eq!(
807            "no auth options are available. This can happen if there's a problem with \
808            the service model, or if there is a codegen bug.",
809            err.to_string()
810        );
811
812        let mut list = ExploredList::default();
813        list.push(
814            AuthSchemeId::new("SigV4"),
815            ExploreResult::NoIdentityResolver,
816        );
817        list.push(
818            AuthSchemeId::new("SigV4a"),
819            ExploreResult::NoIdentityResolver,
820        );
821        let err = NoMatchingAuthSchemeError(list);
822        assert_eq!(
823            "failed to select an auth scheme to sign the request with. \
824            \"SigV4\" wasn't a valid option because there was no identity resolver for it. \
825            \"SigV4a\" wasn't a valid option because there was no identity resolver for it. \
826            Be sure to set an identity, such as credentials, auth token, or other identity \
827            type that is required for this service.",
828            err.to_string()
829        );
830
831        // It should prioritize the suggestion to try an identity before saying it's a bug
832        let mut list = ExploredList::default();
833        list.push(
834            AuthSchemeId::new("SigV4"),
835            ExploreResult::NoIdentityResolver,
836        );
837        list.push(
838            AuthSchemeId::new("SigV4a"),
839            ExploreResult::MissingEndpointConfig,
840        );
841        let err = NoMatchingAuthSchemeError(list);
842        assert_eq!(
843            "failed to select an auth scheme to sign the request with. \
844            \"SigV4\" wasn't a valid option because there was no identity resolver for it. \
845            \"SigV4a\" wasn't a valid option because there is auth config in the endpoint \
846            config, but this scheme wasn't listed in it (see \
847            https://github.com/smithy-lang/smithy-rs/discussions/3281 for more details). \
848            Be sure to set an identity, such as credentials, auth token, or other identity \
849            type that is required for this service.",
850            err.to_string()
851        );
852
853        // Otherwise, it should suggest it's a bug
854        let mut list = ExploredList::default();
855        list.push(
856            AuthSchemeId::new("SigV4a"),
857            ExploreResult::MissingEndpointConfig,
858        );
859        let err = NoMatchingAuthSchemeError(list);
860        assert_eq!(
861            "failed to select an auth scheme to sign the request with. \
862            \"SigV4a\" wasn't a valid option because there is auth config in the endpoint \
863            config, but this scheme wasn't listed in it (see \
864            https://github.com/smithy-lang/smithy-rs/discussions/3281 for more details). \
865            This is likely a bug.",
866            err.to_string()
867        );
868
869        // Truncation should be indicated
870        let mut list = ExploredList::default();
871        for _ in 0..=MAX_EXPLORED_LIST_LEN {
872            list.push(
873                AuthSchemeId::new("dontcare"),
874                ExploreResult::MissingEndpointConfig,
875            );
876        }
877        let err = NoMatchingAuthSchemeError(list).to_string();
878        if !err.contains(
879            "Note: there were other auth schemes that were evaluated that weren't listed here",
880        ) {
881            panic!("The error should indicate that the explored list was truncated.");
882        }
883    }
